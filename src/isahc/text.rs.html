11// This macro abstracts over async and sync decoding, since the implementation
12// of decoding a stream into text is the same.
13macro_rules! decode_reader {
14    ($decoder:expr, $buf:ident, $read:expr) => {{
15        let mut decoder = $decoder;
16        let mut buf = [0; 8192];
17        let mut unread = 0;
18
19        loop {
20            let $buf = &mut buf[unread..];
21            let len = match $read {
22                Ok(0) => break,
23                Ok(len) => len,
24                Err(e) if e.kind() == io::ErrorKind::Interrupted => continue,
25                Err(e) => return Err(e),
26            };
27
28            unread = decoder.push(&buf[..unread + len]).len();
29        }
30
31        Ok(decoder.finish(&buf[..unread]))
32    }};
33}

35decl_future! {
36    /// A future returning a response body decoded as text.
37    pub type TextFuture<R> = impl Future<Output = io::Result<String>> + SendIf<R>;
38}

40/// A streaming text decoder that supports multiple encodings.
41pub(crate) struct Decoder {
42    /// Inner decoder implementation.
43    decoder: encoding_rs::Decoder,
44
45    /// The output string that characters are accumulated to.
46    output: String,
47}
48
49impl Decoder {
50    /// Create a new decoder with the given encoding.
51    pub(crate) fn new(encoding: &'static Encoding) -> Self {
52        Self {
53            decoder: encoding.new_decoder(),
54            output: String::new(),
55        }
56    }
57
58    /// Create a new encoder suitable for decoding the given response.
59    pub(crate) fn for_response<T>(response: &Response<T>) -> Self {
60        if let Some(content_type) = response
61            .content_type()
62            .and_then(|header| header.parse::<mime::Mime>().ok())
63        {
64            if let Some(charset) = content_type.get_param(mime::CHARSET) {
65                if let Some(encoding) =
66                    encoding_rs::Encoding::for_label(charset.as_ref().as_bytes())
67                {
68                    return Self::new(encoding);
69                } else {
70                    tracing::warn!("unknown encoding '{}', falling back to UTF-8", charset);
71                }
72            }
73        }
74
75        Self::new(encoding_rs::UTF_8)
76    }
77
78    /// Consume this decoder to decode text from a given synchronous reader.
79    pub(crate) fn decode_reader(self, mut reader: impl io::Read) -> io::Result<String> {
80        decode_reader!(self, buf, reader.read(buf))
81    }
82
83    /// Consume this decoder to decode text from a given asynchronous reader.
84    pub(crate) fn decode_reader_async<'r, R>(self, mut reader: R) -> TextFuture<'r, R>
85    where
86        R: AsyncRead + Unpin + 'r,
87    {
88        TextFuture::new(async move { decode_reader!(self, buf, reader.read(buf).await) })
89    }
90
91    /// Push additional bytes into the decoder, returning any trailing bytes
92    /// that formed a partial character.
93    pub(crate) fn push<'b>(&mut self, buf: &'b [u8]) -> &'b [u8] {
94        self.decode(buf, false)
95    }
96
97    /// Mark the stream as complete and finish the decoding process, returning
98    /// the resulting string.
99    pub(crate) fn finish(mut self, buf: &[u8]) -> String {
100        self.decode(buf, true);
101        self.output
102    }
103
104    fn decode<'b>(&mut self, mut buf: &'b [u8], last: bool) -> &'b [u8] {
105        loop {
106            let (result, consumed, _) = self.decoder.decode_to_string(buf, &mut self.output, last);
107            buf = &buf[consumed..];
108
109            match result {
110                CoderResult::InputEmpty => break,
111                CoderResult::OutputFull => self
112                    .output
113                    .reserve(self.decoder.max_utf8_buffer_length(buf.len()).unwrap()),
114            }
115        }
116
117        // If last is true, buf should always be fully consumed.
118        if cfg!(debug_assertions) && last {
119            assert_eq!(buf.len(), 0);
120        }
121
122        buf
123    }
124}

126#[cfg(test)]
127mod tests {
128    use super::*;
129    use crate::body::Body;
130
131    static_assertions::assert_impl_all!(TextFuture<'_, &mut Body>: Send);
132
133    #[test]
134    fn utf8_decode() {
135        let mut decoder = Decoder::new(encoding_rs::UTF_8);
136
137        assert_eq!(decoder.push(b"hello"), b"");
138        assert_eq!(decoder.push(b" "), b"");
139        assert_eq!(decoder.finish(b"world"), "hello world");
140    }
141
142    #[test]
143    fn utf16_decode() {
144        let bytes = encoding_rs::UTF_16BE.encode("hello world!").0.into_owned();
145        let mut decoder = Decoder::new(encoding_rs::UTF_8);
146
147        for byte in bytes {
148            decoder.push(&[byte]);
149        }
150
151        assert_eq!(decoder.finish(&[]), "hello world!");
152    }
