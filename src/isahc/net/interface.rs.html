12/// Types implementing this trait can be used as a condition to select a network
13/// interface to use for outgoing connections based on some criteria.
14///
15/// You can combine multiple selectors in some cases using tuples to apply
16/// multiple selectors, such as selecting an interface by both its name and IP
17/// address. All selectors in the tuple must match the interface to be selected.
18///
19/// This trait is sealed and cannot be implemented outside of this crate.
20pub trait Selector {
21    #[doc(hidden)]
22    fn into_interface_string(self, _: Sealed) -> InterfaceString;
23}
24
25/// Use whatever interface the TCP stack finds suitable.
26#[derive(Clone, Copy, Debug, Default)]
27pub struct Any;
28
29impl Selector for Any {
30    fn into_interface_string(self, _: Sealed) -> InterfaceString {
31        InterfaceString(None)
32    }
33}

35/// Bind to an interface by IP address.
36impl Selector for IpAddr {
37    fn into_interface_string(self, _: Sealed) -> InterfaceString {
38        InterfaceString(Some(format!("host!{}", self)))
39    }
40}
41
42/// Bind to an interface by IPv4 address.
43impl Selector for Ipv4Addr {
44    fn into_interface_string(self, _: Sealed) -> InterfaceString {
45        IpAddr::from(self).into_interface_string(Sealed)
46    }
47}
48
49/// Bind to an interface by IPv6 address.
50impl Selector for Ipv6Addr {
51    fn into_interface_string(self, _: Sealed) -> InterfaceString {
52        IpAddr::from(self).into_interface_string(Sealed)
53    }
54}

56/// Selects a network interface to use for outgoing connections based on some
57/// criteria.
58#[derive(Clone, Debug)]
59pub(crate) struct InterfaceString(Option<String>);
60
61impl<T: Selector> From<T> for InterfaceString {
62    fn from(selector: T) -> Self {
63        selector.into_interface_string(Sealed)
64    }
65}
66
67impl SetOpt for InterfaceString {
68    fn set_opt<H>(&self, easy: &mut Easy2<H>) -> Result<(), SetOptError> {
69        match self.0.as_ref() {
70            Some(interface) => easy.interface(interface).map_err(Into::into),
71
72            // Use raw FFI because safe wrapper doesn't let us set to null.
73            None => unsafe {
74                match curl_sys::curl_easy_setopt(easy.raw(), curl_sys::CURLOPT_INTERFACE, 0) {
75                    curl_sys::CURLE_OK => Ok(()),
76                    code => Err(curl::Error::new(code).into()),
77                }
78            },
79        }
80    }
81}

87/// Unix-specific network interface selectors.
88#[cfg(unix)]
89mod unix {
90    use super::*;
91    use std::fmt;
92
93    /// Selects a network interface based on its name (such as `eth0`). This
94    /// selector is not available on Windows as it does not really have names
95    /// for network devices.
96    ///
97    /// # Examples
98    ///
99    /// ```
100    /// # use isahc::net::interface::Name;
101    /// let loopback = Name("lo");
102    /// let wifi = Name("wlan0");
103    /// ```
104    pub struct Name<T: AsRef<str>>(pub T);
105
106    impl<T: AsRef<str> + Clone> Clone for Name<T> {
107        fn clone(&self) -> Self {
108            Self(self.0.clone())
109        }
110    }
111
112    impl<T: AsRef<str>> fmt::Debug for Name<T> {
113        fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
114            let name: &str = self.0.as_ref();
115
116            f.debug_tuple("InterfaceName").field(&name).finish()
117        }
118    }
119
120    impl<T: AsRef<str>> Selector for Name<T> {
121        fn into_interface_string(self, _: Sealed) -> InterfaceString {
122            InterfaceString(Some(format!("if!{}", self.0.as_ref())))
123        }
124    }
125
126    /// Selects a network interface to use for outgoing connections based on
127    /// both its name and current IP address. Both criteria must be satisfied
128    /// for the interface to be selected.
129    impl<T, U> Selector for (Name<T>, U)
130    where
131        T: AsRef<str>,
132        U: Into<IpAddr> + Selector,
133    {
134        fn into_interface_string(self, _: Sealed) -> InterfaceString {
135            InterfaceString(Some(format!(
136                "ifhost!{}!{}",
137                self.0.0.as_ref(),
138                self.1.into()
139            )))
140        }
141    }
142
143    /// Selects a network interface to use for outgoing connections based on
144    /// both its name and current IP address. Both criteria must be satisfied
145    /// for the interface to be selected.
146    impl<T, U> Selector for (T, Name<U>)
147    where
148        T: Into<IpAddr> + Selector,
149        U: AsRef<str>,
150    {
151        fn into_interface_string(self, _: Sealed) -> InterfaceString {
152            (self.1, self.0).into_interface_string(Sealed)
153        }
154    }
155}

157#[cfg(test)]
158mod tests {
159    use super::*;
160
161    #[test]
162    fn test_any_interface() {
163        assert!(Any.into_interface_string(Sealed).0.is_none());
164    }
165
166    #[test]
167    #[cfg(unix)]
168    fn test_interface_name() {
169        let selector = Name("eth0");
170        assert_eq!(selector.into_interface_string(Sealed).0.unwrap(), "if!eth0");
171    }
172
173    #[test]
174    fn test_interface_ip_addr() {
175        let selector = Ipv4Addr::new(192, 168, 1, 1);
176        assert_eq!(
177            selector.into_interface_string(Sealed).0.unwrap(),
178            "host!192.168.1.1"
179        );
180    }
181
182    #[test]
183    #[cfg(unix)]
184    fn test_interface_name_and_ip_addr() {
185        let selector = (Name("eth0"), Ipv4Addr::new(192, 168, 1, 1));
186        assert_eq!(
187            selector.into_interface_string(Sealed).0.unwrap(),
188            "ifhost!eth0!192.168.1.1"
189        );
190    }
