1use http::header::HeaderMap;
2
3/// Extension trait for HTTP requests and responses for accessing common headers
4/// in a typed way.
5///
6/// Eventually this trait can be made public once the types are cleaned up a
7/// bit.
8pub(crate) trait HasHeaders {
9    fn headers(&self) -> &HeaderMap;
10
11    fn content_length(&self) -> Option<u64> {
12        self.headers()
13            .get(http::header::CONTENT_LENGTH)
14            .and_then(|v| v.to_str().ok())
15            .and_then(|v| v.parse().ok())
16    }
17
18    fn content_type(&self) -> Option<&str> {
19        self.headers()
20            .get(http::header::CONTENT_TYPE)
21            .and_then(|v| v.to_str().ok())
22    }
23}
24
25impl HasHeaders for HeaderMap {
26    fn headers(&self) -> &HeaderMap {
27        self
28    }
29}
30
31impl<T> HasHeaders for http::Request<T> {
32    fn headers(&self) -> &HeaderMap {
33        self.headers()
34    }
35}
36
37impl<T> HasHeaders for http::Response<T> {
38    fn headers(&self) -> &HeaderMap {
39        self.headers()
40    }
41}
