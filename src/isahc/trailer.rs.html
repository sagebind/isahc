8/// Holds the current state of a trailer for a response.
9///
10/// This object acts as a shared handle that can be cloned and polled from
11/// multiple threads to wait for and act on the response trailer.
12///
13/// There are two typical workflows for accessing trailer headers:
14///
15/// - If you are consuming the response body and then accessing the headers
16///   afterward, then all trailers are guaranteed to have arrived (if any).
17///   [`Trailer::try_get`] will allow you to access them without extra overhead.
18/// - If you are handling trailers in a separate task, callback, or thread, then
19///   either [`Trailer::wait`] or [`Trailer::wait_async`] will allow you to wait
20///   for the trailer headers to arrive and then handle them.
21///
22/// Note that in either approach, trailer headers are delivered to your
23/// application as a single [`HeaderMap`]; it is not possible to handle
24/// individual headers as they arrive.
25#[derive(Clone, Debug)]
26pub struct Trailer {
27    shared: Arc<Shared>,
28}
29
30#[derive(Debug)]
31struct Shared {
32    headers: OnceLock<HeaderMap>,
33    ready: Event,
34}
35
36impl Trailer {
37    /// Get a populated trailer handle containing no headers.
38    pub(crate) fn empty() -> &'static Self {
39        static EMPTY: OnceLock<Trailer> = OnceLock::new();
40
41        EMPTY.get_or_init(|| Self {
42            shared: Arc::new(Shared {
43                headers: OnceLock::from(HeaderMap::new()),
44                ready: Event::new(),
45            }),
46        })
47    }
48
49    /// Returns true if the trailer has been received (if any).
50    ///
51    /// The trailer will not be received until the body stream associated with
52    /// this response has been fully consumed.
53    #[inline]
54    pub fn is_ready(&self) -> bool {
55        self.try_get().is_some()
56    }
57
58    /// Attempt to get the trailer headers without blocking. Returns `None` if
59    /// the trailer has not been received yet.
60    #[inline]
61    pub fn try_get(&self) -> Option<&HeaderMap> {
62        self.shared.headers.get()
63    }
64
65    /// Block the current thread until the trailer headers arrive, and then
66    /// return them.
67    ///
68    /// This is a blocking operation! If you are writing an asynchronous
69    /// application, then you probably want to use [`Trailer::wait_async`]
70    /// instead.
71    pub fn wait(&self) -> &HeaderMap {
72        loop {
73            // Fast path: If the headers are already set, return them.
74            if let Some(headers) = self.try_get() {
75                return headers;
76            }
77
78            // Headers not set, jump into the slow path by creating a new
79            // listener for the ready event.
80            let listener = self.shared.ready.listen();
81
82            // Double-check that the headers are not set.
83            if let Some(headers) = self.try_get() {
84                return headers;
85            }
86
87            // Otherwise, block until they are set.
88            listener.wait();
89
90            // If we got the notification, then the headers are likely to be
91            // set.
92            if let Some(headers) = self.try_get() {
93                return headers;
94            }
95        }
96    }
97
98    /// Block the current thread until the trailer headers arrive or a timeout
99    /// expires.
100    ///
101    /// If the given timeout expired before the trailer arrived then `None` is
102    /// returned.
103    ///
104    /// This is a blocking operation! If you are writing an asynchronous
105    /// application, then you probably want to use [`Trailer::wait_async`]
106    /// instead.
107    pub fn wait_timeout(&self, timeout: Duration) -> Option<&HeaderMap> {
108        // Fast path: If the headers are already set, return them.
109        if let Some(headers) = self.try_get() {
110            return Some(headers);
111        }
112
113        // Headers not set, jump into the slow path by creating a new listener
114        // for the ready event.
115        let listener = self.shared.ready.listen();
116
117        // Double-check that the headers are not set.
118        if let Some(headers) = self.try_get() {
119            return Some(headers);
120        }
121
122        // Otherwise, block with a timeout.
123        if listener.wait_timeout(timeout).is_some() {
124            self.try_get()
125        } else {
126            None
127        }
128    }
129
130    /// Wait asynchronously until the trailer headers arrive, and then return
131    /// them.
132    pub async fn wait_async(&self) -> &HeaderMap {
133        loop {
134            // Fast path: If the headers are already set, return them.
135            if let Some(headers) = self.try_get() {
136                return headers;
137            }
138
139            // Headers not set, jump into the slow path by creating a new
140            // listener for the ready event.
141            let listener = self.shared.ready.listen();
142
143            // Double-check that the headers are not set.
144            if let Some(headers) = self.try_get() {
145                return headers;
146            }
147
148            // Otherwise, wait asynchronously until they are.
149            listener.await;
150
151            // If we got the notification, then the headers are likely to be
152            // set.
153            if let Some(headers) = self.try_get() {
154                return headers;
155            }
156        }
157    }
158}

160pub(crate) struct TrailerWriter {
161    shared: Arc<Shared>,
162    headers: Option<HeaderMap>,
163}
164
165impl TrailerWriter {
166    pub(crate) fn new() -> Self {
167        Self {
168            shared: Arc::new(Shared {
169                headers: Default::default(),
170                ready: Event::new(),
171            }),
172            headers: Some(HeaderMap::new()),
173        }
174    }
175
176    pub(crate) fn trailer(&self) -> Trailer {
177        Trailer {
178            shared: self.shared.clone(),
179        }
180    }
181
182    pub(crate) fn get_mut(&mut self) -> Option<&mut HeaderMap> {
183        self.headers.as_mut()
184    }
185
186    #[inline]
187    pub(crate) fn flush(&mut self) {
188        if !self.flush_impl() {
189            tracing::warn!("tried to flush trailer multiple times");
190        }
191    }
192
193    fn flush_impl(&mut self) -> bool {
194        if let Some(headers) = self.headers.take() {
195            let _ = self.shared.headers.set(headers);
196
197            // Wake up any calls waiting for the headers.
198            self.shared.ready.notify(usize::max_value());
199
200            true
201        } else {
202            false
203        }
204    }
205}
206
207impl Drop for TrailerWriter {
208    fn drop(&mut self) {
209        self.flush_impl();
210    }
