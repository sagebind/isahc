1//! Configuration options related to SSL/TLS for supporting HTTPS requests.
2//!
3//! # Backends
4//!
5//! Isahc does not implement the TLS protocol itself, but instead uses one of
6//! various available *TLS backends* which implement support. With the default
7//! crate configuration, Isahc will use the target platform's "native" TLS
8//! implementation:
9//!
10//! - Windows: [Schannel]
11//! - macOS and iOS: [Secure Transport]
12//! - Linux and other UNIX-like systems: [OpenSSL] or one of its API-compatible
13//!   forks are usually considered the "default" TLS engine on most Linux
14//!   distributions and are treated as the native backend here. The library is
15//!   dynamically linked to by default.
16//!
17//! Regardless of platform, support for TLS is gated behind the optional `tls`
18//! crate feature, which while enabled by default can be disabled if you don't
19//! need or want to make HTTPS requests. Enabling the crate feature enables this
20//! API module, but does not select which backend to use. To select which
21//! backend to use, additional crate features are provided:
22//!
23//! - `default-tls`: Use whatever Isahc's preferred default TLS engine is for
24//!   the target platform. Currently Isahc defaults to using the platform native
25//!   TLS engine, so this is just an alias for `native-tls`. However, this could
26//!   change in the future. This is the TLS feature that is selected by default.
27//! - `native-tls`: Use the target platform's native TLS engine, as described
28//!   earlier. If you explicitly want to use the native TLS backend regardless
29//!   of what Isahc might default to, you should use this feature.
30//! - `native-tls-static`: Use the target platform's native TLS engine, and
31//!   statically link to it where it makes sense. On Windows and macOS this does
32//!   nothing. On platforms using an OpenSSL-compatible library, attempt to
33//!   statically link to the library.
34//! - `rustls-tls`: Use a statically-linked [rustls], a modern TLS library
35//!   written in Rust.
36//! - `rustls-tls-native-certs`: Use [rustls] along with the
37//!   [rustls-native-certs] library to allow rustls to use the platform's native
38//!   root certificate store.
39//!
40//! If using rustls without native cert support, your application will need to
41//! provide its own certificates to use for verification, as none are included
42//! by default.
43//!
44//! There are pros and cons to different backends, and none are best for all use
45//! cases. For a more in-depth look at the available backends see the [wiki
46//! article on TLS
47//! backends](https://github.com/sagebind/isahc/wiki/TLS-Backends).
48//!
49//! [OpenSSL]: https://www.openssl.org
50//! [rustls]: https://github.com/rustls/rustls
51//! [rustls-native-certs]: https://github.com/rustls/rustls-native-certs
52//! [Schannel]: https://docs.microsoft.com/en-us/windows/win32/com/schannel
53//! [Secure Transport]:
54//!     https://developer.apple.com/documentation/security/secure_transport

82/// A builder for creating a custom SSL/TLS connector configuration.
83#[derive(Debug, Default)]
84#[must_use = "builders have no effect if unused"]
85pub struct TlsConfigBuilder {
86    trust_store: TrustStore,
87    issuer_cert: Option<Certificate>,
88    identity: Option<Identity>,
89    ciphers: Option<String>,
90    min_version: Option<ProtocolVersion>,
91    max_version: Option<ProtocolVersion>,
92    danger_accept_invalid_certs: bool,
93    danger_accept_invalid_hosts: bool,
94    danger_accept_revoked_certs: bool,
95}
96
97impl TlsConfigBuilder {
98    /// Set the certificate store containing trusted root certificates to use
99    /// for validating server certificates.
100    ///
101    /// Root certificates are used for validating the authenticity of a server
102    /// before proceeding with a request. If the server presents a certificate
103    /// that matches the server's information, and is signed by a certificate
104    /// authority either in the root certificate store or is itself trusted by
105    /// another certificate in the store, then the server is considered to be
106    /// legitimate.
107    ///
108    /// The default setting is [`TrustStore::native`].
109    ///
110    /// # Examples
111    ///
112    /// ```
113    /// use isahc::tls::{Certificate, TrustStore, TlsConfig};
114    ///
115    /// let config = TlsConfig::builder()
116    ///     // Use the native certificate store
117    ///     .trust_store(TrustStore::native())
118    ///     // Use a specific certificate bundle file
119    ///     .trust_store(TrustStore::from_file("/etc/certs/cabundle.pem"))
120    ///     // Use custom certs in memory
121    ///     .trust_store(TrustStore::custom([
122    ///         Certificate::from_pem("(some long PEM string)"),
123    ///     ]))
124    ///     // You could even include a certificate bundle in your binary
125    ///     .trust_store(TrustStore::custom([
126    ///         Certificate::from_pem(include_str!("../../tests/certs/isrgrootx1.pem")),
127    ///     ]))
128    ///     .build();
129    /// ```
130    pub fn trust_store(mut self, store: TrustStore) -> Self {
131        self.trust_store = store;
132        self
133    }
134
135    /// Specify a specific CA certificate which server certificates must be
136    /// signed by.
137    ///
138    /// Typically, server certificates are validated by checking whether they
139    /// are signed by a trusted root certificate, or by any intermediate CA
140    /// certificate authorized by a root certificate. If this is not strict
141    /// enough for your use case, you can additionally validate that server
142    /// certificates must be signed by a **specific** CA certificate. This
143    /// method specifies that certificate which servers must match exactly.
144    ///
145    /// By default, no issuer certificate is set.
146    pub fn issuer_certificate(mut self, cert: Certificate) -> Self {
147        self.issuer_cert = Some(cert);
148        self
149    }
150
151    /// Add a custom client certificate to use for client authentication, also
152    /// known as *mutual TLS* (mTLS).
153    ///
154    /// SSL/TLS is often used by the client to verify that the server is
155    /// legitimate (one-way), but with *mutual TLS* (mTLS)  it is _also_ used by
156    /// the server to verify that _you_ are legitimate (two-way). If the server
157    /// asks the client to present an approved certificate before continuing,
158    /// then this sets the certificate chain that will be used to prove
159    /// authenticity.
160    ///
161    /// If a certificate or key format given is not supported by the underlying
162    /// SSL/TLS engine, an error will be returned when attempting to send a
163    /// request using the offending certificate or key.
164    ///
165    /// By default, no identity is set.
166    ///
167    /// # Backend support
168    ///
169    /// Support for mutual TLS varies between the available TLS backends. Here
170    /// are some current limitations of note:
171    ///
172    /// - Schannel and Secure Transport require certificates and private keys to
173    ///   be presented together inside a PKCS #12 archive. This can be an actual
174    ///   archive or one in memory.
175    /// - Mutual TLS with Rustls is not supported at all.
176    ///
177    /// # Examples
178    ///
179    /// ```
180    /// use isahc::tls::{Identity, PrivateKey, TlsConfig};
181    ///
182    /// let config = TlsConfig::builder()
183    ///     .identity(Identity::from_pem_file(
184    ///         "client.pem",
185    ///         PrivateKey::from_pem_file("key.pem", String::from("secret"))
186    ///     ))
187    ///     .build();
188    /// ```
189    pub fn identity(mut self, identity: Identity) -> Self {
190        self.identity = Some(identity);
191        self
192    }
193
194    /// Set the minimum allowed protocol version for secure connections.
195    ///
196    /// If specified, the client will attempt to negotiate secure connections
197    /// using the given protocol version or newer if possible. If a server only
198    /// supports protocols older than the set minimum then the connection will
199    /// be terminated with a
200    /// [`ConnectionFailed`][crate::error::ErrorKind::ConnectionFailed] error.
201    ///
202    /// The default value if not set depends on the TLS backend Isahc is
203    /// configured to use. With most backends on most modern systems the default
204    /// will be TLS v1.2, but if you want to be certain then you may want to set
205    /// a minimum version explicitly.
206    pub fn min_version(mut self, version: ProtocolVersion) -> Self {
207        self.min_version = Some(version);
208        self
209    }
210
211    /// Set a maximum allowed protocol version for secure connections.
212    ///
213    /// If specified, the client will attempt to negotiate secure connections
214    /// using the given protocol version or older if possible. If a server only
215    /// supports protocols newer than the set maximum then the connection will
216    /// be terminated with a
217    /// [`ConnectionFailed`][crate::error::ErrorKind::ConnectionFailed] error.
218    ///
219    /// By default no maximum version is enforced, however the newest version
220    /// that can be used will be dependent on what newest protocols are
221    /// supported by the underlying TLS engine.
222    pub fn max_version(mut self, version: ProtocolVersion) -> Self {
223        self.max_version = Some(version);
224        self
225    }
226
227    /// Set a list of ciphers or cipher suites to allow.
228    ///
229    /// The list of valid cipher names is dependent on the underlying SSL/TLS
230    /// backend in use. You can find an up-to-date list of potential cipher names
231    /// at <https://curl.haxx.se/docs/ssl-ciphers.html>.
232    ///
233    /// The default is unset and will result in using whatever ciphers the
234    /// configured TLS backend chooses to enable by default.
235    ///
236    /// # Warning
237    ///
238    /// Not all cipher suites are created equal, as many have been marked as
239    /// being insecure over time. This method allows you to enable potentially
240    /// insecure cipher suites that are not enabled by default and may introduce
241    /// security vulnerabilities into your application.
242    pub fn ciphers<I, T>(mut self, ciphers: I) -> Self
243    where
244        I: IntoIterator<Item = T>,
245        T: AsRef<str>,
246    {
247        let mut iter = ciphers.into_iter();
248
249        // If an empty list is provided, reset to default. Otherwise build up a
250        // string in curl format containing the cipher names.
251        if let Some(first) = iter.next() {
252            let mut ciphers = first.as_ref().to_owned();
253
254            for cipher in iter {
255                ciphers.push(':');
256                ciphers.push_str(cipher.as_ref());
257            }
258
259            self.ciphers = Some(ciphers);
260        } else {
261            self.ciphers = None;
262        }
263
264        self
265    }
266
267    /// Disables all server certificate validation. Connections will be accepted
268    /// from servers that present invalid or expired certificates.
269    ///
270    /// By default this is disabled, and invalid certificates are rejected.
271    ///
272    /// # Warning
273    ///
274    /// **You should think very carefully before using this method.** If invalid
275    /// certificates are trusted, *any* certificate for any site will be trusted
276    /// for use. This includes expired certificates. This introduces significant
277    /// vulnerabilities, and should only be used as a last resort.
278    #[cfg(feature = "tls-insecure")]
279    pub fn danger_accept_invalid_certs(mut self, accept: bool) -> Self {
280        self.danger_accept_invalid_certs = accept;
281        self
282    }
283
284    /// Disables hostname verification on server certificates. Connections will
285    /// be accepted from servers that present certificates that do not match
286    /// their hostname.
287    ///
288    /// By default this is disabled, and connections with servers that return a
289    /// certificate that does not match their hostname are rejected.
290    ///
291    /// # Backend support
292    ///
293    /// This option is **not supported** when using the Rustls TLS backend and
294    /// will have no effect.
295    ///
296    /// # Warning
297    ///
298    /// **You should think very carefully before you use this method.** If
299    /// hostname verification is not used, any valid certificate for any site
300    /// will be trusted for use from any other. This introduces a significant
301    /// vulnerability to man-in-the-middle attacks.
302    #[cfg(feature = "tls-insecure")]
303    pub fn danger_accept_invalid_hosts(mut self, accept: bool) -> Self {
304        self.danger_accept_invalid_hosts = accept;
305        self
306    }
307
308    /// Disables certificate revocation checks for backends where such behavior
309    /// is present.
310    ///
311    /// By default this is disabled, and revoked certificates are rejected.
312    ///
313    /// # Backend support
314    ///
315    /// This option is **only supported** when using the Schannel TLS backend
316    /// (the native Windows SSL/TLS implementation). On other backends this
317    /// option will likely have no effect regardless of setting.
318    ///
319    /// # Warning
320    ///
321    /// **You should think very carefully before you use this method.** Backends
322    /// implementing revocation checks do so to ensure that a certificate that
323    /// appears valid has not been reported as compromised. Disabling this can
324    /// increase your application's vulnerability to malicious servers.
325    #[cfg(feature = "tls-insecure")]
326    pub fn danger_accept_revoked_certs(mut self, accept: bool) -> Self {
327        self.danger_accept_revoked_certs = accept;
328        self
329    }
330
331    /// Create a new [`TlsConfig`] based on this builder.
332    pub fn build(self) -> TlsConfig {
333        TlsConfig {
334            curl_flags: {
335                let mut options = SslOpt::new();
336                options.no_revoke(self.danger_accept_revoked_certs);
337                self.trust_store.configure_ssl_options(&mut options);
338                options
339            },
340            trust_store: self.trust_store,
341            issuer_cert: self.issuer_cert,
342            identity: self.identity,
343            ciphers: self.ciphers,
344            min_version: self
345                .min_version
346                .as_ref()
347                .map(ProtocolVersion::as_curl_ssl_version),
348            max_version: self
349                .max_version
350                .as_ref()
351                .map(ProtocolVersion::as_curl_ssl_version),
352            danger_accept_invalid_certs: self.danger_accept_invalid_certs,
353            danger_accept_invalid_hosts: self.danger_accept_invalid_hosts,
354        }
355    }
356}

358/// Configuration for making SSL/TLS connections.
359#[derive(Clone, Debug)]
360pub struct TlsConfig {
361    trust_store: TrustStore,
362    issuer_cert: Option<Certificate>,
363    identity: Option<Identity>,
364
365    /// List of ciphers to use, in a string format compatible with curl.
366    ciphers: Option<String>,
367    min_version: Option<SslVersion>,
368    max_version: Option<SslVersion>,
369    danger_accept_invalid_certs: bool,
370    danger_accept_invalid_hosts: bool,
371    curl_flags: SslOpt,
372}
373
374impl TlsConfig {
375    /// Create an instance of the default SSL/TLS configuration.
376    ///
377    /// The default configuration varies depending on the runtime environment
378    /// and how Isahc is compiled. By default, Isahc uses the platform's
379    /// "native" TLS implementation, and will use TLS ciphers and versions that
380    /// are enabled by default on the system. Naturally, this will vary from
381    /// machine to machine, so less-secure methods may be used if the user's
382    /// system is configured that way.
383    ///
384    /// If Isahc is built to use an alternative TLS implementation such as
385    /// rustls, then that implementation will be used on all platforms, and the
386    /// default settings will come from the defaults provided by that bundled
387    /// TLS library version.
388    ///
389    /// This is equivalent to the [`Default::default`] method.
390    pub fn new() -> Self {
391        Self::builder().build()
392    }
393
394    /// Create a new [`TlsConfigBuilder`] for creating a custom SSL/TLS
395    /// configuration. The initial configuration of the builder will start from
396    /// the default TLS settings as described in [`TlsConfig::builder`].
397    #[inline]
398    pub fn builder() -> TlsConfigBuilder {
399        TlsConfigBuilder::default()
400    }
401}

409impl SetOpt for TlsConfig {
410    fn set_opt<H>(&self, easy: &mut Easy2<H>) -> Result<(), SetOptError> {
411        if let Some(ciphers) = self.ciphers.as_ref() {
412            easy.ssl_cipher_list(ciphers)?;
413        }
414
415        easy.ssl_options(&self.curl_flags)?;
416        easy.ssl_verify_peer(!self.danger_accept_invalid_certs)?;
417        easy.ssl_verify_host(!self.danger_accept_invalid_hosts)?;
418
419        // Rustls only supports TLS 1.2+. Currently the backend in curl just
420        // ignores this option entirely, so give a more helpful message if the
421        // user tries to explicitly use something older.
422        if TlsEngine::Rustls.is_available()
423            && !matches!(
424                self.max_version,
425                Some(SslVersion::Tlsv12)
426                    | Some(SslVersion::Tlsv13)
427                    | Some(SslVersion::Default)
428                    | None
429            )
430        {
431            return Err(Error::new(
432                ErrorKind::TlsEngine,
433                MaximumTlsVersionNotSupportedByEngineError {
434                    max_requested: self.max_version.unwrap(),
435                    engine: TlsEngine::Rustls,
436                },
437            )
438            .into());
439        }
440
441        easy.ssl_min_max_version(
442            self.min_version.unwrap_or(SslVersion::Default),
443            self.max_version.unwrap_or(SslVersion::Default),
444        )?;
445
446        self.trust_store.set_opt(easy)?;
447
448        if let Some(cert) = self.issuer_cert.as_ref() {
449            easy.issuer_cert_blob(cert.as_pem_bytes())?;
450        }
451
452        if let Some(identity) = self.identity.as_ref() {
453            identity.set_opt(easy)?;
454        }
455
456        Ok(())
457    }
458}

460impl SetOptProxy for TlsConfig {
461    fn set_opt_proxy<H>(&self, easy: &mut Easy2<H>) -> Result<(), SetOptError> {
462        if let Some(ciphers) = self.ciphers.as_ref() {
463            easy.proxy_ssl_cipher_list(ciphers)?;
464        }
465
466        easy.proxy_ssl_options(&self.curl_flags)?;
467        easy.proxy_ssl_verify_peer(self.danger_accept_invalid_certs)?;
468        easy.proxy_ssl_verify_host(self.danger_accept_invalid_hosts)?;
469
470        easy.proxy_ssl_min_max_version(
471            self.min_version.unwrap_or(SslVersion::Default),
472            self.max_version.unwrap_or(SslVersion::Default),
473        )?;
474
475        self.trust_store.set_opt_proxy(easy)?;
476
477        if let Some(cert) = self.issuer_cert.as_ref() {
478            easy.proxy_issuer_cert_blob(cert.as_pem_bytes())?;
479        }
480
481        if let Some(identity) = self.identity.as_ref() {
482            identity.set_opt_proxy(easy)?;
483        }
484
485        Ok(())
486    }
487}

489/// Possible SSL/TLS versions that can be used.
490///
491/// Not all versions are supported by all TLS backends, and can vary depending
492/// how a user's system is configured. Requesting or requiring a version that
493/// the TLS backend does not allow or support will result in a runtime error.
494#[derive(Clone, Copy, Debug, PartialEq, Eq)]
495#[non_exhaustive]
496pub enum ProtocolVersion {
497    /// SSL v2. Support for this version is disabled in the default build
498    /// configuration and not even available any more in most TLS backends due
499    /// to being considered insecure.
500    Sslv2,
501
502    /// SSL v3. Support for this version is disabled in the default build
503    /// configuration and not even available any more in most TLS backends due
504    /// to being considered insecure.
505    Sslv3,
506
507    /// TLS v1.0. Considered insecure.
508    Tlsv10,
509
510    /// TLS v1.1. Considered insecure.
511    Tlsv11,
512
513    /// TLS v1.2
514    Tlsv12,
515
516    /// TLS v1.3
517    Tlsv13,
518}
519
520impl ProtocolVersion {
521    const fn as_curl_ssl_version(&self) -> SslVersion {
522        match self {
523            Self::Sslv2 => SslVersion::Sslv2,
524            Self::Sslv3 => SslVersion::Sslv3,
525            Self::Tlsv10 => SslVersion::Tlsv10,
526            Self::Tlsv11 => SslVersion::Tlsv11,
527            Self::Tlsv12 => SslVersion::Tlsv12,
528            Self::Tlsv13 => SslVersion::Tlsv13,
529        }
530    }
531}

533/// Known TLS engines that curl might be compiled with for detection at runtime.
534#[derive(Clone, Copy, Debug, PartialEq, Eq)]
535#[non_exhaustive]
536enum TlsEngine {
537    Rustls,
538    Schannel,
539    SecureTransport,
540}
541
542impl TlsEngine {
543    fn is_available(&self) -> bool {
544        if let Some(version) = curl_info().ssl_version() {
545            match self {
546                TlsEngine::Rustls => version.contains("rustls/"),
547                TlsEngine::Schannel => version.contains("Schannel"),
548                TlsEngine::SecureTransport => version.contains("SecureTransport"),
549            }
550        } else {
551            false
552        }
553    }
554}

556/// A highly specific error type that is returned when a user attempts to set a
557/// maximum TLS version that is known to not be supported by the configured TLS
558/// engine.
559#[derive(Clone, Debug)]
560struct MaximumTlsVersionNotSupportedByEngineError {
561    max_requested: SslVersion,
562    engine: TlsEngine,
563}
564
565impl std::error::Error for MaximumTlsVersionNotSupportedByEngineError {}
566
567impl fmt::Display for MaximumTlsVersionNotSupportedByEngineError {
568    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
569        write!(
570            f,
571            "TLS version {:?} is not supported by the {:?} TLS engine",
572            self.max_requested, self.engine
573        )
574    }
