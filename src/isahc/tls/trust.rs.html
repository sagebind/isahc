1//! This module provides the API used to configure how trust of TLS server
2//! certificates and certificate authorities should be determined.
3//!
4//! This API is called "trust" because it might not be as simple as a list of
5//! root certificates. You can delegate to other verifier APIs to determine
6//! which certificates you trust, for example.

17/// A store that provides a collection of trusted root certificates.
18///
19/// Root certificates are used for validating the authenticity of a server
20/// before proceeding with a request. If the server presents a certificate that
21/// matches the server's information, and is signed by a certificate authority
22/// either in the root certificate store or is itself trusted by another
23/// certificate in the store, then the server is considered to be legitimate.
24///
25/// Isahc supports multiple kinds of stores, though the default is to use the
26/// shared store provided by the operating system (if any).
27#[derive(Clone, Debug)]
28pub struct TrustStore(Repr);
29
30#[derive(Clone, Debug)]
31enum Repr {
32    /// Do nothing to configure trust, and rely on curl's default behavior.
33    NoOp,
34
35    // Unset CA-related options in case they are set by default. Usually this is
36    // a way of asking to use the OS root certificate store for certain
37    // backends.
38    Unset,
39
40    /// Sets the `CURLSSLOPT_NATIVE_CA` flag, which asks curl to use the OS
41    /// native CA store, if possible and supported by the current TLS backend.
42    ///
43    /// When using rustls, this will cause curl to use the
44    /// rustls-platform-verifier crate (via rustls-ffi) to verify any certs.
45    ///
46    /// See <https://curl.se/libcurl/c/CURLOPT_SSL_OPTIONS.html>.
47    NativeCa,
48
49    /// Use a certificate bundle from a file path.
50    ///
51    /// This may or may not be combined with OS-native certificate stores,
52    /// depending on the TLS backend.
53    FilePath(PathBuf),
54
55    /// Use an in-memory bundle of certificates.
56    ///
57    /// This may or may not be combined with OS-native certificate stores,
58    /// depending on the TLS backend.
59    PemBundle(String),
60}
61
62impl TrustStore {
63    /// Use the operating system's native APIs for verifying certificate trust,
64    /// if possible.
65    ///
66    /// This is normally the trust method used for most typical applications.
67    /// This is also returned as the [`Default`] implementation.
68    ///
69    /// On Windows, macOS, and iOS this involves using the certificate
70    /// management features provided by the operating system. On Linux and other
71    /// UNIX-like systems this typically will use a shared certificate bundle
72    /// managed by the distribution or system administrator. In most cases, this
73    /// will also respect environment variables that override where to look for
74    /// trusted certificates.
75    ///
76    /// # Error handling
77    ///
78    /// The presence or ability to access a system certificate store is not
79    /// checked here. If the system store cannot be accessed due to permissions
80    /// or some other kind of problem, an error will be returned when attempting
81    /// to send a request using the store.
82    ///
83    /// If the system store is simply empty or at least *appears* to be empty,
84    /// the TLS backend will probably not consider this an inherent error,
85    /// though naturally you will likely encounter certificate errors since no
86    /// certificates will be considered trusted.
87    pub fn native() -> Self {
88        /// To determine how to access the native store we have to perform some
89        /// runtime checks and probing, so we only do this once and cache the
90        /// result.
91        static NATIVE_STORE: LazyLock<TrustStore> = LazyLock::new(TrustStore::new_native);
92
93        NATIVE_STORE.clone()
94    }
95
96    fn new_native() -> Self {
97        // Ensure curl (and if applicable, openssl-probe) are initialized
98        // before doing anything.
99        curl::init();
100
101        // If the `SSL_CERT_FILE` environment variable is set, use that. At
102        // least in a unix-like environment, doing so is considered common
103        // courtesy.
104        //
105        // Note: OpenSSL checks `SSL_CERT_FILE` by default so we wouldn't need
106        // to do this, but LibreSSL doesn't. This makes the behavior consistent
107        // between the two. It also means that if openssl-probe (which the curl
108        // crate may run during initialization) discovered a cert file to use
109        // using its discovery mechanism, we will use it even with LibreSSL even
110        // though openssl-probe doesn't work with LibreSSL out of the box.
111        if let Some(path) = env::var_os("SSL_CERT_FILE") {
112            if !path.is_empty() {
113                tracing::debug!(
114                    ?path,
115                    "using certificate bundle from SSL_CERT_FILE environment variable",
116                );
117                return TrustStore::from_file(path);
118            }
119        }
120
121        // If we are using curl 8.13.0+ with rustls, then we can ask for
122        // rustls-platform-verifier to be used to verify server certificates
123        // using the OS-native certificate store.
124        if TlsEngine::Rustls.is_available() && curl_version() >= (8, 13, 0) {
125            tracing::debug!("using platform verifier with rustls");
126            return TrustStore(Repr::NativeCa);
127        }
128
129        // These backends will use the store built into the OS as long as we
130        // ensure no paths are set. They shouldn't be when curl is statically
131        // linked, but they might be if using a system curl.
132        if TlsEngine::Schannel.is_available() || TlsEngine::SecureTransport.is_available() {
133            return TrustStore(Repr::Unset);
134        }
135
136        // If all else fails, use whatever curl wants to fall back to, if any.
137        TrustStore(Repr::NoOp)
138    }
139
140    /// Use a file containing a bundle of certificates in PEM format.
141    ///
142    /// The certificate bundle is not loaded or validated here. If the file does
143    /// not exist or the format is not supported by the underlying SSL/TLS
144    /// engine, an error will be returned when attempting to send a request
145    /// using the offending bundle.
146    ///
147    /// # Examples
148    ///
149    /// ```
150    /// use isahc::tls::TrustStore;
151    ///
152    /// let store = TrustStore::from_file("/etc/certs/cabundle.pem");
153    /// ```
154    pub fn from_file<P: Into<PathBuf>>(path: P) -> Self {
155        Self(Repr::FilePath(path.into()))
156    }
157
158    /// Create a custom certificate store containing exactly the given
159    /// certificates.
160    ///
161    /// Server certificates will be verified using only certificates given.
162    ///
163    /// This store is supported by most TLS backends, including OpenSSL, rustls,
164    /// and Secure Transport.
165    pub fn custom<I>(certificates: I) -> Self
166    where
167        I: IntoIterator<Item = Certificate>,
168    {
169        // Generate a PEM bundle.
170        let bundle = certificates
171            .into_iter()
172            .map(|cert| cert.into_pem_string())
173            .collect();
174
175        Self(Repr::PemBundle(bundle))
176    }
177
178    pub(super) fn configure_ssl_options(&self, ssl_opt: &mut SslOpt) {
179        if let Repr::NativeCa = &self.0 {
180            ssl_opt.native_ca(true);
181        }
182    }
183}
184
185impl Default for TrustStore {
186    fn default() -> Self {
187        Self::native()
188    }
189}

191impl SetOpt for TrustStore {
192    fn set_opt<H>(&self, easy: &mut Easy2<H>) -> Result<(), SetOptError> {
193        match &self.0 {
194            Repr::NativeCa | Repr::NoOp => Ok(()),
195            Repr::FilePath(path) => easy.cainfo(path).map_err(Into::into),
196            Repr::PemBundle(bundle) => easy.ssl_cainfo_blob(bundle.as_bytes()).map_err(Into::into),
197            Repr::Unset => {
198                // safe wrapper does not allow setting to null
199                unsafe {
200                    curl_sys::curl_easy_setopt(
201                        easy.raw(),
202                        curl_sys::CURLOPT_CAINFO,
203                        ptr::null_mut::<c_char>(),
204                    );
205                    curl_sys::curl_easy_setopt(
206                        easy.raw(),
207                        curl_sys::CURLOPT_CAPATH,
208                        ptr::null_mut::<c_char>(),
209                    );
210                }
211
212                Ok(())
213            }
214        }
215    }
216}

218impl SetOptProxy for TrustStore {
219    fn set_opt_proxy<H>(&self, easy: &mut Easy2<H>) -> Result<(), SetOptError> {
220        match &self.0 {
221            Repr::NativeCa | Repr::NoOp => Ok(()),
222            Repr::FilePath(path) => {
223                if let Some(path) = path.to_str() {
224                    easy.proxy_cainfo(path).map_err(Into::into)
225                } else {
226                    Err(Error::new(
227                        ErrorKind::InvalidTlsConfiguration,
228                        CertificatePathNotUtf8Error { path: path.clone() },
229                    )
230                    .into())
231                }
232            }
233            Repr::PemBundle(bundle) => easy
234                .proxy_ssl_cainfo_blob(bundle.as_bytes())
235                .map_err(Into::into),
236            Repr::Unset => {
237                // safe wrapper does not allow setting to null
238                unsafe {
239                    curl_sys::curl_easy_setopt(
240                        easy.raw(),
241                        curl_sys::CURLOPT_PROXY_CAINFO,
242                        ptr::null_mut::<c_char>(),
243                    );
244                    curl_sys::curl_easy_setopt(
245                        easy.raw(),
246                        curl_sys::CURLOPT_PROXY_CAPATH,
247                        ptr::null_mut::<c_char>(),
248                    );
249                }
250
251                Ok(())
252            }
253        }
254    }
255}

257/// An X.509 digital certificate.
258#[derive(Clone, Debug)]
259pub struct Certificate {
260    /// Curl prefers to work in the PEM format, so internally we do as well.
261    pem: String,
262}
263
264impl Certificate {
265    /// Use one or more DER-encoded certificates stored in memory.
266    ///
267    /// The certificates are not parsed or validated here. If a certificate is
268    /// malformed or the format is not supported by the underlying SSL/TLS
269    /// engine, an error will be returned when attempting to send a request
270    /// using the offending certificate.
271    pub fn from_der<B: AsRef<[u8]>>(der: B) -> Self {
272        #[cfg(windows)]
273        const LINE_ENDING: &str = "\r\n";
274        #[cfg(not(windows))]
275        const LINE_ENDING: &str = "\n";
276
277        let mut base64_spec = data_encoding::BASE64.specification();
278        base64_spec.wrap.width = 64;
279        base64_spec.wrap.separator.push_str(LINE_ENDING);
280        let base64 = base64_spec.encoding().unwrap();
281
282        let mut pem = String::new();
283
284        pem.push_str("-----BEGIN CERTIFICATE-----");
285        pem.push_str(LINE_ENDING);
286        base64.encode_append(der.as_ref(), &mut pem);
287        pem.push_str("-----END CERTIFICATE-----");
288        pem.push_str(LINE_ENDING);
289        Self::from_pem(pem)
290    }
291
292    /// Use one or more PEM-encoded certificates in the given byte buffer.
293    ///
294    /// The certificate object takes ownership of the byte buffer. If a borrowed
295    /// type is supplied, such as `&[u8]`, then the bytes will be copied.
296    ///
297    /// The certificates are not parsed or validated here. If a certificate is
298    /// malformed or the format is not supported by the underlying SSL/TLS
299    /// engine, an error will be returned when attempting to send a request
300    /// using the offending certificate.
301    pub fn from_pem<B: AsRef<[u8]>>(pem: B) -> Self {
302        Self {
303            pem: String::from_utf8(pem.as_ref().to_vec()).unwrap(),
304        }
305    }
306
307    pub(crate) fn as_pem_bytes(&self) -> &[u8] {
308        self.pem.as_bytes()
309    }
310
311    pub(crate) fn into_pem_string(self) -> String {
312        self.pem
313    }
314}

316#[derive(Clone, Debug)]
317struct CertificatePathNotUtf8Error {
318    path: PathBuf,
319}
320
321impl std::error::Error for CertificatePathNotUtf8Error {}
322
323impl fmt::Display for CertificatePathNotUtf8Error {
324    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
325        write!(
326            f,
327            "certificate path is not valid UTF-8: {}",
328            self.path.display()
329        )
330    }
331}
