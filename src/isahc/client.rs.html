47/// An HTTP client builder, capable of creating custom [`HttpClient`] instances
48/// with customized behavior.
49///
50/// Any option that can be configured per-request can also be configured on a
51/// client builder as a default setting. Request configuration is provided by
52/// the [`Configurable`] trait, which is also available in the
53/// [`prelude`](crate::prelude) module.
54///
55/// # Examples
56///
57/// ```
58/// use isahc::{
59///     config::{RedirectPolicy, VersionNegotiation},
60///     prelude::*,
61///     HttpClient,
62/// };
63/// use std::time::Duration;
64///
65/// let client = HttpClient::builder()
66///     .timeout(Duration::from_secs(60))
67///     .redirect_policy(RedirectPolicy::Limit(10))
68///     .version_negotiation(VersionNegotiation::http2())
69///     .build()?;
70/// # Ok::<(), isahc::Error>(())
71/// ```
72#[must_use = "builders have no effect if unused"]
73pub struct HttpClientBuilder {
74    agent_builder: AgentBuilder,
75    client_config: ClientConfig,
76    request_config: RequestConfig,
77    interceptors: Vec<InterceptorObj>,
78    default_headers: HeaderMap<HeaderValue>,
79    error: Option<Error>,
80
81    #[cfg(feature = "cookies")]
82    cookie_jar: Option<crate::cookies::CookieJar>,
83}
84

86    fn default() -> Self {
87        Self::new()
88    }

173    /// Set a maximum number of simultaneous connections that this client is
174    /// allowed to keep open at one time.
175    ///
176    /// If set to a value greater than zero, no more than `max` connections will
177    /// be opened at one time. If executing a new request would require opening
178    /// a new connection, then the request will stay in a "pending" state until
179    /// an existing connection can be used or an active request completes and
180    /// can be closed, making room for a new connection.
181    ///
182    /// Setting this value to `0` disables the limit entirely.
183    ///
184    /// This is an effective way of limiting the number of sockets or file
185    /// descriptors that this client will open, though note that the client may
186    /// use file descriptors for purposes other than just HTTP connections.
187    ///
188    /// By default this value is `0` and no limit is enforced.
189    ///
190    /// To apply a limit per-host, see
191    /// [`HttpClientBuilder::max_connections_per_host`].
192    pub fn max_connections(mut self, max: usize) -> Self {
193        self.agent_builder = self.agent_builder.max_connections(max);
194        self
195    }

250    /// Configure DNS caching.
251    ///
252    /// By default, DNS entries are cached by the client executing the request
253    /// and are used until the entry expires. Calling this method allows you to
254    /// change the entry timeout duration or disable caching completely.
255    ///
256    /// Note that DNS entry TTLs are not respected, regardless of this setting.
257    ///
258    /// By default caching is enabled with a 60 second timeout.
259    ///
260    /// # Examples
261    ///
262    /// ```
263    /// use isahc::{config::*, prelude::*, HttpClient};
264    /// use std::time::Duration;
265    ///
266    /// let client = HttpClient::builder()
267    ///     // Cache entries for 10 seconds.
268    ///     .dns_cache(Duration::from_secs(10))
269    ///     // Cache entries forever.
270    ///     .dns_cache(DnsCache::Forever)
271    ///     // Don't cache anything.
272    ///     .dns_cache(DnsCache::Disable)
273    ///     .build()?;
274    /// # Ok::<(), isahc::Error>(())
275    /// ```
276    pub fn dns_cache<C>(mut self, cache: C) -> Self
277    where
278        C: Into<DnsCache>,
279    {
280        // This option is technically supported per-request by curl, but we only
281        // expose it on the client. Since the DNS cache is shared between all
282        // requests, exposing this option per-request would actually cause the
283        // timeout to alternate values for every request with a different
284        // timeout, resulting in some confusing (but valid) behavior.
285        self.client_config.dns_cache = Some(cache.into());
286        self
287    }

290    ///
291    /// Entries in the given map will be used first before using the default DNS
292    /// resolver for host+port pairs.
293    ///
294    /// Note that DNS resolving is only performed when establishing a new

317    /// Add a default header to be passed with every request.
318    ///
319    /// If a default header value is already defined for the given key, then a
320    /// second header value will be appended to the list and multiple header
321    /// values will be included in the request.
322    ///
323    /// If any values are defined for this header key on an outgoing request,
324    /// they will override any default header values.
325    ///
326    /// If the header key or value are malformed, [`HttpClientBuilder::build`]
327    /// will return an error.
328    ///
329    /// # Examples
330    ///
331    /// ```
332    /// use isahc::{prelude::*, HttpClient};
333    ///
334    /// let client = HttpClient::builder()
335    ///     .default_header("some-header", "some-value")
336    ///     .build()?;
337    /// # Ok::<(), Box<dyn std::error::Error>>(())
338    /// ```
339    pub fn default_header<K, V>(mut self, key: K, value: V) -> Self
340    where
341        HeaderName: TryFrom<K>,
342        <HeaderName as TryFrom<K>>::Error: Into<http::Error>,
343        HeaderValue: TryFrom<V>,
344        <HeaderValue as TryFrom<V>>::Error: Into<http::Error>,
345    {
346        match HeaderName::try_from(key) {
347            Ok(key) => match HeaderValue::try_from(value) {
348                Ok(value) => {
349                    self.default_headers.append(key, value);
350                }
351                Err(e) => {
352                    self.error = Some(Error::new(ErrorKind::ClientInitialization, e.into()));
353                }
354            },
355            Err(e) => {
356                self.error = Some(Error::new(ErrorKind::ClientInitialization, e.into()));
357            }
358        }
359        self
360    }

362    /// Set the default headers to include in every request, replacing any
363    /// previously set default headers.
364    ///
365    /// Headers defined on an individual request always override headers in the
366    /// default map.
367    ///
368    /// If any header keys or values are malformed, [`HttpClientBuilder::build`]
369    /// will return an error.
370    ///
371    /// # Examples
372    ///
373    /// Set default headers from a slice:
374    ///
375    /// ```
376    /// use isahc::{prelude::*, HttpClient};
377    ///
378    /// let mut builder = HttpClient::builder()
379    ///     .default_headers(&[
380    ///         ("some-header", "value1"),
381    ///         ("some-header", "value2"),
382    ///         ("some-other-header", "some-other-value"),
383    ///     ])
384    ///     .build()?;
385    /// # Ok::<(), Box<dyn std::error::Error>>(())
386    /// ```
387    ///
388    /// Using an existing header map:
389    ///
390    /// ```
391    /// use isahc::{prelude::*, HttpClient};
392    ///
393    /// let mut headers = http::HeaderMap::new();
394    /// headers.append("some-header".parse::<http::header::HeaderName>()?, "some-value".parse()?);
395    ///
396    /// let mut builder = HttpClient::builder()
397    ///     .default_headers(&headers)
398    ///     .build()?;
399    /// # Ok::<(), Box<dyn std::error::Error>>(())
400    /// ```
401    ///
402    /// Using a hashmap:
403    ///
404    /// ```
405    /// use isahc::{prelude::*, HttpClient};
406    /// use std::collections::HashMap;
407    ///
408    /// let mut headers = HashMap::new();
409    /// headers.insert("some-header", "some-value");
410    ///
411    /// let mut builder = HttpClient::builder()
412    ///     .default_headers(headers)
413    ///     .build()?;
414    /// # Ok::<(), Box<dyn std::error::Error>>(())
415    /// ```
416    pub fn default_headers<K, V, I, P>(mut self, headers: I) -> Self
417    where
418        HeaderName: TryFrom<K>,
419        <HeaderName as TryFrom<K>>::Error: Into<http::Error>,
420        HeaderValue: TryFrom<V>,
421        <HeaderValue as TryFrom<V>>::Error: Into<http::Error>,
422        I: IntoIterator<Item = P>,
423        P: HeaderPair<K, V>,
424    {
425        self.default_headers.clear();
426
427        for (key, value) in headers.into_iter().map(HeaderPair::pair) {
428            self = self.default_header(key, value);
429        }
430
431        self
432    }

434    /// Build an [`HttpClient`] using the configured options.
435    ///
436    /// If the client fails to initialize, an error will be returned.
437    #[allow(unused_mut)]
438    pub fn build(mut self) -> Result<HttpClient, Error> {
439        if let Some(err) = self.error {
440            return Err(err);
441        }
442
443        // Add cookie interceptor if enabled.
444        #[cfg(feature = "cookies")]
445        {
446            let jar = self.cookie_jar.clone();
447            self = self.interceptor_impl(crate::cookies::interceptor::CookieInterceptor::new(jar));
448        }
449
450        // Add default header interceptor if any default headers were specified.
451        if !self.default_headers.is_empty() {
452            let default_headers = std::mem::take(&mut self.default_headers);
453            self = self.interceptor_impl(DefaultHeadersInterceptor::from(default_headers));
454        }
455
456        #[cfg(not(feature = "cookies"))]
457        let inner = Inner {
458            agent: self
459                .agent_builder
460                .spawn()
461                .map_err(|e| Error::new(ErrorKind::ClientInitialization, e))?,
462            client_config: self.client_config,
463            request_config: self.request_config,
464            interceptors: self.interceptors,
465        };
466
467        #[cfg(feature = "cookies")]
468        let inner = Inner {
469            agent: self
470                .agent_builder
471                .spawn()
472                .map_err(|e| Error::new(ErrorKind::ClientInitialization, e))?,
473            client_config: self.client_config,
474            request_config: self.request_config,
475            interceptors: self.interceptors,
476            cookie_jar: self.cookie_jar,
477        };
478
479        Ok(HttpClient {
480            inner: Arc::new(inner),
481        })
482    }
483}

527/// An HTTP client for making requests.
528///
529/// An [`HttpClient`] instance acts as a session for executing one or more HTTP
530/// requests, and also allows you to set common protocol settings that should be
531/// applied to all requests made with the client.
532///
533/// [`HttpClient`] is entirely thread-safe, and implements both [`Send`] and
534/// [`Sync`]. You are free to create clients outside the context of the "main"
535/// thread, or move them between threads. You can even invoke many requests
536/// simultaneously from multiple threads, since doing so doesn't need a mutable
537/// reference to the client. This is fairly cheap to do as well, since
538/// internally requests use lock-free message passing to get things going.
539///
540/// The client maintains a connection pool internally and is not cheap to
541/// create, so we recommend creating a client once and re-using it throughout
542/// your code. Creating a new client for every request would decrease
543/// performance significantly, and might cause errors to occur under high
544/// workloads, caused by creating too many system resources like sockets or
545/// threads.
546///
547/// It is not universally true that you should use exactly one client instance
548/// in an application. All HTTP requests made with the same client will share
549/// any session-wide state, like cookies or persistent connections. It may be
550/// the case that it is better to create separate clients for separate areas of
551/// an application if they have separate concerns or are making calls to
552/// different servers. If you are creating an API client library, that might be
553/// a good place to maintain your own internal client.
554///
555/// # Examples
556///
557/// ```no_run
558/// use isahc::{prelude::*, HttpClient};
559///
560/// // Create a new client using reasonable defaults.
561/// let client = HttpClient::new()?;
562///
563/// // Make some requests.
564/// let mut response = client.get("https://example.org")?;
565/// assert!(response.status().is_success());
566///
567/// println!("Response:\n{}", response.text()?);
568/// # Ok::<(), isahc::Error>(())
569/// ```
570///
571/// Customizing the client configuration:
572///
573/// ```no_run
574/// use isahc::{
575///     config::{RedirectPolicy, VersionNegotiation},
576///     prelude::*,
577///     HttpClient,
578/// };
579/// use std::time::Duration;
580///
581/// let client = HttpClient::builder()
582///     .version_negotiation(VersionNegotiation::http11())
583///     .redirect_policy(RedirectPolicy::Limit(10))
584///     .timeout(Duration::from_secs(20))
585///     // May return an error if there's something wrong with our configuration
586///     // or if the client failed to start up.
587///     .build()?;
588///
589/// let response = client.get("https://example.org")?;
590/// assert!(response.status().is_success());
591/// # Ok::<(), isahc::Error>(())
592/// ```
593///
594/// See the documentation on [`HttpClientBuilder`] for a comprehensive look at
595/// what can be configured.
596#[derive(Clone)]
597pub struct HttpClient {
598    inner: Arc<Inner>,
599}

863    /// Send an HTTP request and return the HTTP response.
864    ///
865    /// Upon success, will return a [`Response`] containing the status code,
866    /// response headers, and response body from the server. The [`Response`] is
867    /// returned as soon as the HTTP response headers are received; the
868    /// connection will remain open to stream the response body in real time.
869    /// Dropping the response body without fully consuming it will close the
870    /// connection early without downloading the rest of the response body.
871    ///
872    /// The response body is provided as a stream that may only be consumed
873    /// once. If you need to inspect the response body more than once, you will
874    /// have to either read it into memory or write it to a file.
875    ///
876    /// The response body is not a direct stream from the server, but uses its
877    /// own buffering mechanisms internally for performance. It is therefore
878    /// undesirable to wrap the body in additional buffering readers.
879    ///
880    /// _Note that the actual underlying socket connection isn't necessarily
881    /// closed on drop. It may remain open to be reused if pipelining is being
882    /// used, the connection is configured as `keep-alive`, and so on._
883    ///
884    /// This client's configuration can be overridden for this request by
885    /// configuring the request using methods provided by the [`Configurable`]
886    /// trait.
887    ///
888    /// To execute a request asynchronously, see [`HttpClient::send_async`].
889    ///
890    /// # Examples
891    ///
892    /// ```no_run
893    /// use isahc::{prelude::*, HttpClient, Request};
894    ///
895    /// let client = HttpClient::new()?;
896    ///
897    /// let request = Request::post("https://httpbin.org/post")
898    ///     .header("Content-Type", "application/json")
899    ///     .body(r#"{
900    ///         "speed": "fast",
901    ///         "cool_name": true
902    ///     }"#)?;
903    ///
904    /// let response = client.send(request)?;
905    /// assert!(response.status().is_success());
906    /// # Ok::<(), isahc::Error>(())
907    /// ```
908    pub fn send<B>(&self, request: Request<B>) -> Result<Response<Body>, Error>
909    where
910        B: Into<Body>,
911    {
912        let span = tracing::debug_span!(
913            "send",
914            method = ?request.method(),
915            uri = ?request.uri(),
916        );
917
918        let mut writer_maybe = None;
919
920        let request = request.map(|body| {
921            let (async_body, writer) = body.into().into_async();
922            writer_maybe = writer;
923            async_body
924        });
925
926        let response = block_on(
927            async move {
928                // Instead of simply blocking the current thread until the response
929                // is received, we can use the current thread to read from the
930                // request body synchronously while concurrently waiting for the
931                // response.
932                if let Some(mut writer) = writer_maybe {
933                    // Note that the `send_async` future is given first; this
934                    // ensures that it is polled first and thus the request is
935                    // initiated before we attempt to write the request body.
936                    let (response, _) = try_zip(self.send_async_inner(request), async move {
937                        writer.write().await.map_err(Error::from)
938                    })
939                    .await?;
940
941                    Ok(response)
942                } else {
943                    self.send_async_inner(request).await
944                }
945            }
946            .instrument(span),
947        )?;
948
949        Ok(response.map(|body| body.into_sync()))
950    }

954    /// Upon success, will return a [`Response`] containing the status code,
955    /// response headers, and response body from the server. The [`Response`] is
956    /// returned as soon as the HTTP response headers are received; the
957    /// connection will remain open to stream the response body in real time.

959    /// connection early without downloading the rest of the response body.

965    /// The response body is not a direct stream from the server, but uses its
966    /// own buffering mechanisms internally for performance. It is therefore
967    /// undesirable to wrap the body in additional buffering readers.
968    ///
969    /// _Note that the actual underlying socket connection isn't necessarily
970    /// closed on drop. It may remain open to be reused if pipelining is being
971    /// used, the connection is configured as `keep-alive`, and so on._
972    ///
973    /// This client's configuration can be overridden for this request by
974    /// configuring the request using methods provided by the [`Configurable`]
975    /// trait.
976    ///
977    /// To execute a request synchronously, see [`HttpClient::send`].

979    /// # Examples

999    pub fn send_async<B>(&self, request: Request<B>) -> ResponseFuture<'_>
1000    where
1001        B: Into<AsyncBody>,
1002    {
1003        let span = tracing::debug_span!(
1004            "send_async",
1005            method = ?request.method(),
1006            uri = ?request.uri(),
1007        );
1008
1009        ResponseFuture::new(
1010            self.send_async_inner(request.map(Into::into))
1011                .instrument(span),
1012        )
1013    }

1015    /// Actually send the request. All the public methods go through here.
1016    async fn send_async_inner(
1017        &self,
1018        mut request: Request<AsyncBody>,
1019    ) -> Result<Response<AsyncBody>, Error> {
1020        // Populate request config, creating if necessary.
1021        if let Some(config) = request.extensions_mut().get_mut::<RequestConfig>() {
1022            // Merge request configuration with defaults.
1023            config.merge(&self.inner.request_config);
1024        } else {
1025            request
1026                .extensions_mut()
1027                .insert(self.inner.request_config.clone());
1028        }
1029
1030        let ctx = interceptor::Context {
1031            invoker: Arc::new(self),
1032            interceptors: &self.inner.interceptors,
1033        };
1034
1035        ctx.send(request).await
1036    }

1038    fn create_easy_handle(
1039        &self,
1040        mut request: Request<AsyncBody>,
1041    ) -> Result<
1042        (
1043            curl::easy::Easy2<RequestHandler>,
1044            impl Future<Output = Result<Response<ResponseBodyReader>, Error>>,
1045        ),
1046        SetOptError,
1047    > {
1048        // Prepare the request plumbing.
1049        let body = std::mem::take(request.body_mut());
1050        let has_body = !body.is_empty();
1051        let body_length = body.len();
1052        let (handler, future) = RequestHandler::new(body);
1053
1054        let mut easy = curl::easy::Easy2::new(handler);
1055
1056        // Set whether curl should generate verbose debug data for us to log.
1057        easy.verbose(easy.get_ref().is_debug_enabled())?;
1058
1059        // Disable connection reuse logs if connection cache is disabled.
1060        if self.inner.client_config.close_connections {
1061            easy.get_mut().disable_connection_reuse_log = true;
1062        }
1063
1064        easy.signal(false)?;
1065
1066        let request_config = request.extensions().get::<RequestConfig>().unwrap();
1067
1068        request_config.set_opt(&mut easy)?;
1069        self.inner.client_config.set_opt(&mut easy)?;
1070
1071        // Check if we need to disable the Expect header.
1072        let disable_expect_header = request_config
1073            .expect_continue
1074            .as_ref()
1075            .map(|x| x.is_disabled())
1076            .unwrap_or_default();
1077
1078        // Set the HTTP method to use. Curl ties in behavior with the request
1079        // method, so we need to configure this carefully.
1080        match (request.method(), has_body) {
1081            // Normal GET request.
1082            (&http::Method::GET, false) => {
1083                easy.get(true)?;
1084            }
1085            // Normal HEAD request.
1086            (&http::Method::HEAD, false) => {
1087                easy.nobody(true)?;
1088            }
1089            // POST requests have special redirect behavior.
1090            (&http::Method::POST, true) => {
1091                easy.post(true)?;
1092            }
1093            // Normal PUT request.
1094            (&http::Method::PUT, true) => {
1095                easy.upload(true)?;
1096            }
1097            // Default case is to either treat request like a GET or PUT.
1098            (method, has_body) => {
1099                easy.upload(has_body)?;
1100                easy.custom_request(method.as_str())?;
1101            }
1102        }
1103
1104        easy.url(&uri_to_string(request.uri()))?;
1105
1106        // If the request has a body, then we either need to tell curl how large
1107        // the body is if we know it, or tell curl to use chunked encoding. If
1108        // we do neither, curl will simply not send the body without warning.
1109        if has_body {
1110            // Use length given in Content-Length header, or the size defined by
1111            // the body itself.
1112            let body_length = request
1113                .headers()
1114                .get("Content-Length")
1115                .and_then(|value| value.to_str().ok())
1116                .and_then(|value| value.parse().ok())
1117                .or(body_length);
1118
1119            if let Some(len) = body_length {
1120                if request.method() == http::Method::POST {
1121                    easy.post_field_size(len)?;
1122                } else {
1123                    easy.in_filesize(len)?;
1124                }
1125            } else {
1126                // Set the Transfer-Encoding header to instruct curl to use
1127                // chunked encoding. Replaces any existing values that may be
1128                // incorrect.
1129                request.headers_mut().insert(
1130                    "Transfer-Encoding",
1131                    http::header::HeaderValue::from_static("chunked"),
1132                );
1133            }
1134        }
1135
1136        // Generate a header list for curl.
1137        let mut headers = curl::easy::List::new();
1138
1139        let title_case = request
1140            .extensions()
1141            .get::<RequestConfig>()
1142            .unwrap()
1143            .title_case_headers
1144            .unwrap_or(false);
1145
1146        for (name, value) in request.headers().iter() {
1147            headers.append(&header_to_curl_string(name, value, title_case))?;
1148        }
1149
1150        if disable_expect_header {
1151            headers.append("Expect:")?;
1152        }
1153
1154        easy.http_headers(headers)?;
1155
1156        Ok((easy, future))
1157    }
1158}

1160impl crate::interceptor::Invoke for &HttpClient {
1161    fn invoke(
1162        &self,
1163        mut request: Request<AsyncBody>,
1164    ) -> crate::interceptor::InterceptorFuture<'_, Error> {
1165        Box::pin(async move {
1166            let is_head_request = request.method() == http::Method::HEAD;
1167
1168            // Set default user agent if not specified.
1169            request
1170                .headers_mut()
1171                .entry(http::header::USER_AGENT)
1172                .or_insert(USER_AGENT.parse().unwrap());
1173
1174            // Check if automatic decompression is enabled; we'll need to know
1175            // this later after the response is sent.
1176            let is_automatic_decompression = request
1177                .extensions()
1178                .get::<RequestConfig>()
1179                .unwrap()
1180                .automatic_decompression
1181                .unwrap_or(false);
1182
1183            // Create and configure a curl easy handle to fulfil the request.
1184            let (easy, future) = self.create_easy_handle(request)?;
1185
1186            // Send the request to the agent to be executed.
1187            self.inner.agent.submit_request(easy)?;
1188
1189            // Await for the response headers.
1190            let response = future.await?;
1191
1192            // If a Content-Length header is present, include that information in
1193            // the body as well.
1194            let body_len = response.content_length().filter(|_| {
1195                // If automatic decompression is enabled, and will likely be
1196                // selected, then the value of Content-Length does not indicate
1197                // the uncompressed body length and merely the compressed data
1198                // length. If it looks like we are in this scenario then we
1199                // ignore the Content-Length, since it can only cause confusion
1200                // when included with the body.
1201                if is_automatic_decompression {
1202                    if let Some(value) = response.headers().get(http::header::CONTENT_ENCODING) {
1203                        if value != "identity" {
1204                            return false;
1205                        }
1206                    }
1207                }
1208
1209                true
1210            });
1211
1212            // Convert the reader into an opaque Body.
1213            Ok(response.map(|reader| {
1214                if is_head_request {
1215                    AsyncBody::empty()
1216                } else {
1217                    let body = ResponseBody {
1218                        inner: reader,
1219                        // Extend the lifetime of the agent by including a reference
1220                        // to its handle in the response body.
1221                        _client: (*self).clone(),
1222                    };
1223
1224                    if let Some(len) = body_len {
1225                        AsyncBody::from_reader_sized(body, len)
1226                    } else {
1227                        AsyncBody::from_reader(body)
1228                    }
1229                }
1230            }))
1231        })
1232    }
1233}

1290/// Convert a URI to a string. This implementation is a bit faster than the
1291/// `Display` implementation that avoids the `std::fmt` machinery.
1292fn uri_to_string(uri: &http::Uri) -> String {
1293    let mut s = String::new();
1294
1295    if let Some(scheme) = uri.scheme() {
1296        s.push_str(scheme.as_str());
1297        s.push_str("://");
1298    }
1299
1300    if let Some(authority) = uri.authority() {
1301        s.push_str(authority.as_str());
1302    }
1303
1304    s.push_str(uri.path());
1305
1306    if let Some(query) = uri.query() {
1307        s.push('?');
1308        s.push_str(query);
1309    }
1310
1311    s
1312}
