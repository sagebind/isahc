1use super::{
2    dns::{DnsCache, ResolveMap},
3    setopt::{SetOpt, SetOptError},

5use std::time::Duration;
6
7#[derive(Debug, Default)]
8pub(crate) struct ClientConfig {
9    pub(crate) connection_cache_ttl: Option<Duration>,
10    pub(crate) close_connections: bool,
11    pub(crate) dns_cache: Option<DnsCache>,
12    pub(crate) dns_resolve: Option<ResolveMap>,
13}
14
15impl SetOpt for ClientConfig {
16    fn set_opt<H>(&self, easy: &mut curl::easy::Easy2<H>) -> Result<(), SetOptError> {
17        if let Some(ttl) = self.connection_cache_ttl {
18            easy.maxage_conn(ttl)?;
19        }
20
21        if let Some(cache) = self.dns_cache.as_ref() {
22            cache.set_opt(easy)?;
23        }
24
25        if let Some(map) = self.dns_resolve.as_ref() {
26            map.set_opt(easy)?;
27        }
28
29        easy.forbid_reuse(self.close_connections)?;
30
31        Ok(())
32    }
33}
