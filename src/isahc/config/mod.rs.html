1//! Definition of all client and request configuration options.
2//!
3//! Individual options are separated out into multiple types. Each type acts
4//! both as a "field name" and the value of that option.
5
6// Options are implemented as structs of various kinds that can be "applied" to
7// a curl easy handle. This helps to reduce code duplication as there are many
8// supported options, and also helps avoid having a massive function that does
9// all the configuring.
10//
11// When adding new config options, remember to add methods for setting the
12// option both in HttpClientBuilder and RequestBuilderExt. In addition, be sure
13// to update the client code to apply the option when configuring an easy
14// handle.
15

36/// Provides additional methods when building a request for configuring various
37/// execution-related options on how the request should be sent.
38///
39/// This trait can be used to either configure requests individually by invoking
40/// them on an [`http::request::Builder`], or to configure the default settings
41/// for an [`HttpClient`](crate::HttpClient) by invoking them on an
42/// [`HttpClientBuilder`](crate::HttpClientBuilder).
43///
44/// This trait is sealed and cannot be implemented for types outside of Isahc.
45pub trait Configurable: request::WithRequestConfig {
46    /// Specify a maximum amount of time that a complete request/response cycle
47    /// is allowed to take before being aborted. This includes DNS resolution,
48    /// connecting to the server, writing the request, and reading the response.
49    ///
50    /// Since response bodies are streamed, you will likely receive a
51    /// [`Response`](crate::http::Response) before the response body stream has
52    /// been fully consumed. This means that the configured timeout will still
53    /// be active for that request, and if it expires, further attempts to read
54    /// from the stream will return a [`TimedOut`](std::io::ErrorKind::TimedOut)
55    /// I/O error.
56    ///
57    /// This also means that if you receive a response with a body but do not
58    /// immediately start reading from it, then the timeout timer will still be
59    /// active and may expire before you even attempt to read the body. Keep
60    /// this in mind when consuming responses and consider handling the response
61    /// body right after you receive it if you are using this option.
62    ///
63    /// If not set, no timeout will be enforced.
64    ///
65    /// # Examples
66    ///
67    /// ```no_run
68    /// use isahc::{prelude::*, Request};
69    /// use std::time::Duration;
70    ///
71    /// // This page is too slow and won't respond in time.
72    /// let response = Request::get("https://httpbin.org/delay/10")
73    ///     .timeout(Duration::from_secs(5))
74    ///     .body(())?
75    ///     .send()
76    ///     .expect_err("page should time out");
77    /// # Ok::<(), isahc::Error>(())
78    /// ```
79    #[must_use = "builders have no effect if unused"]
80    fn timeout(self, timeout: Duration) -> Self {
81        self.with_config(move |config| {
82            config.timeout = Some(timeout);
83        })
84    }

213    /// Configure the use of the `Expect` request header when sending request
214    /// bodies with HTTP/1.1.
215    ///
216    /// By default, when sending requests containing a body of large or unknown
217    /// length over HTTP/1.1, Isahc will send the request headers first without
218    /// the body and wait for the server to respond with a 100 (Continue) status
219    /// code, as defined by [RFC 7231, Section
220    /// 5.1.1](https://datatracker.ietf.org/doc/html/rfc7231#section-5.1.1).
221    /// This gives the opportunity for the server to reject the response without
222    /// needing to first transmit the request body over the network, if the body
223    /// contents are not necessary for the server to determine an appropriate
224    /// response.
225    ///
226    /// For servers that do not support this behavior and instead simply wait
227    /// for the request body without responding with a 100 (Continue), there is
228    /// a limited timeout before the response body is sent anyway without
229    /// confirmation. The default timeout is 1 second, but this can be
230    /// configured.
231    ///
232    /// The `Expect` behavior can also be disabled entirely.
233    ///
234    /// This configuration only takes effect when using HTTP/1.1.
235    ///
236    /// # Examples
237    ///
238    /// ```
239    /// use std::time::Duration;
240    /// use isahc::{
241    ///     config::ExpectContinue,
242    ///     prelude::*,
243    ///     HttpClient,
244    /// };
245    ///
246    /// // Use the default behavior (enabled).
247    /// let client = HttpClient::builder()
248    ///     .expect_continue(ExpectContinue::enabled())
249    ///     // or equivalently...
250    ///     .expect_continue(ExpectContinue::default())
251    ///     // or equivalently...
252    ///     .expect_continue(true)
253    ///     .build()?;
254    ///
255    /// // Customize the timeout if the server doesn't respond with a 100
256    /// // (Continue) status.
257    /// let client = HttpClient::builder()
258    ///     .expect_continue(ExpectContinue::timeout(Duration::from_millis(200)))
259    ///     // or equivalently...
260    ///     .expect_continue(Duration::from_millis(200))
261    ///     .build()?;
262    ///
263    /// // Disable the Expect header entirely.
264    /// let client = HttpClient::builder()
265    ///     .expect_continue(ExpectContinue::disabled())
266    ///     // or equivalently...
267    ///     .expect_continue(false)
268    ///     .build()?;
269    /// # Ok::<(), isahc::Error>(())
270    /// ```
271    #[must_use = "builders have no effect if unused"]
272    fn expect_continue<T>(self, expect: T) -> Self
273    where
274        T: Into<ExpectContinue>,
275    {
276        self.with_config(move |config| {
277            config.expect_continue = Some(expect.into());
278        })
279    }

337    /// Configure which network interface local sockets should be bound to for sending
338    /// network traffic from.
339    ///
340    /// Note that interface selection is not performed until you actually
341    /// attempt to send a request, so the validity of your selection is not
342    /// checked here.
343    ///
344    /// # Examples
345    ///
346    /// Bind to an IP address.
347    ///
348    /// ```
349    /// use isahc::{prelude::*, HttpClient};
350    /// use std::net::IpAddr;
351    ///
352    /// let client = HttpClient::builder()
353    ///     .interface(IpAddr::from([192, 168, 1, 2]))
354    ///     .build()?;
355    /// # Ok::<(), isahc::Error>(())
356    /// ```
357    ///
358    /// Bind to an interface by name (not supported on Windows).
359    ///
360    /// ```
361    /// # #[cfg(unix)] {
362    /// use isahc::{prelude::*, net::interface::Name, HttpClient};
363    ///
364    /// let client = HttpClient::builder()
365    ///     .interface(Name("eth0"))
366    ///     .build()?;
367    /// # }
368    /// # Ok::<(), isahc::Error>(())
369    /// ```
370    ///
371    /// Bind to an interface by both name and IP address.
372    ///
373    /// ```
374    /// # #[cfg(unix)] {
375    /// use isahc::{prelude::*, net::interface::Name, HttpClient};
376    /// use std::net::IpAddr;
377    ///
378    /// let client = HttpClient::builder()
379    ///     .interface((Name("eth0"), IpAddr::from([192, 168, 1, 2])))
380    ///     .build()?;
381    /// # }
382    /// # Ok::<(), isahc::Error>(())
383    /// ```
384    ///
385    /// Reset to using whatever interface the TCP stack finds suitable (the
386    /// default).
387    ///
388    /// ```
389    /// use isahc::{prelude::*, net::interface::Any, Request};
390    ///
391    /// let request = Request::get("https://example.org")
392    ///     .interface(Any)
393    ///     .body(())?;
394    /// # Ok::<(), isahc::Error>(())
395    /// ```
396    #[must_use = "builders have no effect if unused"]
397    fn interface<I>(self, interface: I) -> Self
398    where
399        I: interface::Selector,
400    {
401        self.with_config(move |config| {
402            config.interface = Some(interface.into());
403        })
404    }

468    /// Set a proxy to use for requests.
469    ///
470    /// The proxy protocol is specified by the URI scheme.
471    ///
472    /// - **`http`**: Proxy. Default when no scheme is specified.
473    /// - **`https`**: HTTPS Proxy. (Added in 7.52.0 for OpenSSL, GnuTLS and
474    ///   NSS)
475    /// - **`socks4`**: SOCKS4 Proxy.
476    /// - **`socks4a`**: SOCKS4a Proxy. Proxy resolves URL hostname.
477    /// - **`socks5`**: SOCKS5 Proxy.
478    /// - **`socks5h`**: SOCKS5 Proxy. Proxy resolves URL hostname.
479    ///
480    /// By default no proxy will be used, unless one is specified in either the
481    /// `http_proxy` or `https_proxy` environment variables.
482    ///
483    /// Setting to `None` explicitly disables the use of a proxy.
484    ///
485    /// # Examples
486    ///
487    /// Using `http://proxy:80` as a proxy:
488    ///
489    /// ```
490    /// use isahc::{prelude::*, HttpClient};
491    ///
492    /// let client = HttpClient::builder()
493    ///     .proxy(Some("http://proxy:80".parse()?))
494    ///     .build()?;
495    /// # Ok::<(), Box<dyn std::error::Error>>(())
496    /// ```
497    ///
498    /// Explicitly disable the use of a proxy:
499    ///
500    /// ```
501    /// use isahc::{prelude::*, HttpClient};
502    ///
503    /// let client = HttpClient::builder()
504    ///     .proxy(None)
505    ///     .build()?;
506    /// # Ok::<(), Box<dyn std::error::Error>>(())
507    /// ```
508    #[must_use = "builders have no effect if unused"]
509    fn proxy(self, proxy: impl Into<Option<http::Uri>>) -> Self {
510        self.with_config(move |config| {
511            config.proxy = Some(proxy.into());
512        })
513    }

601    /// Set various options for this request that control SSL/TLS behavior.
602    ///
603    /// Some options are for disabling security checks that introduce security
604    /// risks, but may be required as a last resort. Note that the most secure
605    /// options are typically the default and do not need to be specified.
606    ///
607    /// The default value is
608    /// [`TlsConfig::default`](crate::tls::TlsConfig::default).
609    ///
610    /// # Warning
611    ///
612    /// You should think very carefully before using this method. Using *any*
613    /// options that alter how certificates are validated might introduce
614    /// significant security vulnerabilities.
615    ///
616    /// # Examples
617    ///
618    /// ```no_run
619    /// use isahc::{prelude::*, tls::{TlsConfig, TrustStore}, Request};
620    ///
621    /// let response = Request::get("https://example.org")
622    ///     .tls_config(TlsConfig::builder()
623    ///         .trust_store(TrustStore::native())
624    ///         .build())
625    ///     .body(())?
626    ///     .send()?;
627    /// # Ok::<(), isahc::Error>(())
628    /// ```
629    ///
630    /// ```no_run
631    /// use isahc::{prelude::*, tls::TlsConfig, Request};
632    ///
633    /// # #[cfg(feature = "tls-insecure")]
634    /// let response = Request::get("https://badssl.com")
635    ///     .tls_config(TlsConfig::builder()
636    ///         .danger_accept_invalid_certs(true)
637    ///         .danger_accept_revoked_certs(true)
638    ///         .build())
639    ///     .body(())?
640    ///     .send()?;
641    /// # Ok::<(), isahc::Error>(())
642    /// ```
643    ///
644    /// ```
645    /// use isahc::{prelude::*, tls::TlsConfig, HttpClient};
646    ///
647    /// # #[cfg(feature = "tls-insecure")]
648    /// let client = HttpClient::builder()
649    ///     .tls_config(TlsConfig::builder()
650    ///         .danger_accept_invalid_certs(true)
651    ///         .danger_accept_revoked_certs(true)
652    ///         .build())
653    ///     .build()?;
654    /// # Ok::<(), isahc::Error>(())
655    /// ```
656    #[cfg(feature = "tls")]
657    #[must_use = "builders have no effect if unused"]
658    fn tls_config(self, tls_config: crate::tls::TlsConfig) -> Self {
659        self.with_config(move |config| {
660            config.tls_config = Some(tls_config);
661        })

757/// A strategy for selecting what HTTP versions should be used when
758/// communicating with a server.
759///
760/// You can set a version negotiation strategy on a given request or on a client
761/// with [`Configurable::version_negotiation`].
762///
763/// Attempting to use an HTTP version without client-side support at runtime
764/// will result in an error. For example, using the system libcurl on an old
765/// machine may not have an HTTP/2 implementation. Using static linking and the
766/// [`http2`](../index.html#http2) crate feature can help guarantee that HTTP/2
767/// will be available to use.
768#[derive(Clone, Debug)]
769pub struct VersionNegotiation(VersionNegotiationInner);
770
771#[derive(Clone, Copy, Debug)]
772enum VersionNegotiationInner {
773    LatestCompatible,
774    Strict(curl::easy::HttpVersion),
775}
776
777impl Default for VersionNegotiation {
778    fn default() -> Self {
779        Self::latest_compatible()
780    }
781}
782
783impl VersionNegotiation {
784    /// Always prefer the latest supported version announced by the server,
785    /// falling back to older versions if not explicitly listed as supported.
786    /// This is the default.
787    ///
788    /// Secure connections will begin with a TLS handshake, after which the
789    /// highest supported HTTP version listed by the server via ALPN will be
790    /// used. Once connected, additional upgrades to newer versions may also
791    /// occur if the server lists support for it. In the future, headers such as
792    /// `Alt-Svc` will be used.
793    ///
794    /// Insecure connections always use HTTP/1.x since there is no standard
795    /// mechanism for a server to declare support for insecure HTTP versions,
796    /// and only HTTP/0.9, HTTP/1.x, and HTTP/2 support insecure transfers.
797    pub const fn latest_compatible() -> Self {
798        Self(VersionNegotiationInner::LatestCompatible)
799    }
800
801    /// Connect via HTTP/1.0 and do not attempt to use a higher version.
802    pub const fn http10() -> Self {
803        Self(VersionNegotiationInner::Strict(
804            curl::easy::HttpVersion::V10,
805        ))
806    }
807
808    /// Connect via HTTP/1.1 and do not attempt to use a higher version.
809    pub const fn http11() -> Self {
810        Self(VersionNegotiationInner::Strict(
811            curl::easy::HttpVersion::V11,
812        ))
813    }
814
815    /// Connect via HTTP/2. Failure to connect will not fall back to old
816    /// versions, unless HTTP/1.1 is negotiated via TLS ALPN before the session
817    /// begins.
818    ///
819    /// If HTTP/2 support is not compiled in, then using this strategy will
820    /// always result in an error.
821    ///
822    /// This strategy is often referred to as [HTTP/2 with Prior
823    /// Knowledge](https://http2.github.io/http2-spec/#known-http).
824    pub const fn http2() -> Self {
825        Self(VersionNegotiationInner::Strict(
826            curl::easy::HttpVersion::V2PriorKnowledge,
827        ))
828    }
829
830    /// Connect via HTTP/3. Failure to connect will not fall back to old
831    /// versions.
832    pub const fn http3() -> Self {
833        Self(VersionNegotiationInner::Strict(curl::easy::HttpVersion::V3))
834    }
835}
836
837impl SetOpt for VersionNegotiation {
838    fn set_opt(&self, easy: &mut EasyHandle) -> Result<(), SetOptError> {
839        match self.0 {
840            VersionNegotiationInner::LatestCompatible => {
841                // If HTTP/2 support is available, this basically the most
842                // lenient way of using it. Alt-Svc is used to upgrade to newer
843                // versions, and old versions are used if the server doesn't
844                // list HTTP/2 via ALPN.
845                //
846                // If HTTP/2 is not available, leaving it the default setting is
847                // the ideal behavior.
848                if is_http_version_supported(http::Version::HTTP_2) {
849                    easy.http_version(curl::easy::HttpVersion::V2TLS)
850                        .map_err(Into::into)
851                } else {
852                    Ok(())
853                }
854            }
855            VersionNegotiationInner::Strict(version) => {
856                easy.http_version(version).map_err(Into::into)
857            }
858        }
859    }
860}

862/// Supported IP versions that can be used.
863#[derive(Clone, Debug)]
864pub enum IpVersion {
865    /// Use IPv4 addresses only. IPv6 addresses will be ignored.
866    V4,
867
868    /// Use IPv6 addresses only. IPv4 addresses will be ignored.
869    V6,
870
871    /// Use either IPv4 or IPv6 addresses. By default IPv6 addresses are
872    /// preferred if available, otherwise an IPv4 address will be used. IPv6
873    /// addresses are tried first by following the recommendations of [RFC
874    /// 6555 "Happy Eyeballs"](https://tools.ietf.org/html/rfc6555).
875    Any,
876}

881    }

893    }

896/// Controls the use of the `Expect` request header when sending request bodies
897/// with HTTP/1.1.
898///
899/// By default, when sending requests containing a body of large or unknown
900/// length over HTTP/1.1, Isahc will send the request headers first without the
901/// body and wait for the server to respond with a 100 (Continue) status code,
902/// as defined by [RFC 7231, Section
903/// 5.1.1](https://datatracker.ietf.org/doc/html/rfc7231#section-5.1.1). This
904/// gives the opportunity for the server to reject the response without needing
905/// to first transmit the request body over the network, if the body contents
906/// are not necessary for the server to determine an appropriate response.
907///
908/// For servers that do not support this behavior and instead simply wait for
909/// the request body without responding with a 100 (Continue), there is a
910/// limited timeout before the response body is sent anyway without
911/// confirmation. The default timeout is 1 second, but this can be configured.
912///
913/// The `Expect` behavior can also be disabled entirely.
914///
915/// This configuration only takes effect when using HTTP/1.1.
916#[derive(Clone, Debug)]
917pub struct ExpectContinue {
918    timeout: Option<Duration>,
919}
920
921impl ExpectContinue {
922    /// Enable the use of `Expect` and wait for a 100 (Continue) response with a
923    /// default timeout before sending a request body.
924    pub const fn enabled() -> Self {
925        Self::timeout(Duration::from_secs(1))
926    }
927
928    /// Enable the use of `Expect` and wait for a 100 (Continue) response, up to
929    /// the given timeout, before sending a request body.
930    pub const fn timeout(timeout: Duration) -> Self {
931        Self {
932            timeout: Some(timeout),
933        }
934    }
935
936    /// Disable the use and handling of the `Expect` request header.
937    pub const fn disabled() -> Self {
938        Self { timeout: None }
939    }
940
941    pub(crate) fn is_disabled(&self) -> bool {
942        self.timeout.is_none()
943    }
944}
945
946impl Default for ExpectContinue {
947    fn default() -> Self {
948        Self::enabled()
949    }
950}
951
952impl From<bool> for ExpectContinue {
953    fn from(value: bool) -> Self {
954        if value {
955            Self::enabled()
956        } else {
957            Self::disabled()
958        }
959    }
960}
961
962impl From<Duration> for ExpectContinue {
963    fn from(value: Duration) -> Self {
964        Self::timeout(value)
965    }
966}
967
968impl SetOpt for ExpectContinue {
969    fn set_opt(&self, easy: &mut EasyHandle) -> Result<(), SetOptError> {
970        if let Some(timeout) = self.timeout {
971            easy.expect_100_timeout(timeout)?;
972        }
973
974        Ok(())
975    }
