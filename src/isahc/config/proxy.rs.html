1use super::setopt::{EasyHandle, SetOpt, SetOptError};
2use std::iter::FromIterator;
3
4/// A list of host names that do not require a proxy to get reached, even if one
5/// is specified.
6///
7/// See
8/// [`HttpClientBuilder::proxy_blacklist`](crate::HttpClientBuilder::proxy_blacklist)
9/// for configuring a client's no proxy list.
10#[derive(Clone, Debug)]
11pub(crate) struct Blacklist {
12    skip: String,
13}
14
15impl FromIterator<String> for Blacklist {
16    fn from_iter<I: IntoIterator<Item = String>>(iter: I) -> Self {
17        Self {
18            skip: iter.into_iter().collect::<Vec<_>>().join(","),
19        }
20    }
21}
22
23impl SetOpt for Blacklist {
24    fn set_opt(&self, easy: &mut EasyHandle) -> Result<(), SetOptError> {
25        easy.noproxy(&self.skip)?;
26        Ok(())
27    }
28}
