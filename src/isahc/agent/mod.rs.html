1//! Curl agent that executes multiple requests simultaneously.
2//!
3//! The agent is implemented as a single background thread attached to a
4//! "handle". The handle communicates with the agent thread by using message
5//! passing. The agent executes multiple curl requests simultaneously by using a
6//! single "multi" handle.
7//!
8//! Since request executions are driven through futures, the agent also acts as
9//! a specialized task executor for tasks related to requests.
10

32static NEXT_AGENT_ID: AtomicCell<usize> = AtomicCell::new(0);
33const WAIT_TIMEOUT: Duration = Duration::from_millis(1000);

37/// Builder for configuring and spawning an agent.
38#[derive(Debug, Default)]
39pub(crate) struct AgentBuilder {
40    max_connections: usize,
41    max_connections_per_host: usize,
42    connection_cache_size: usize,
43}

45impl AgentBuilder {
46    pub(crate) fn max_connections(mut self, max: usize) -> Self {
47        self.max_connections = max;
48        self
49    }

52        self.max_connections_per_host = max;

61    /// Spawn a new agent using the configuration in this builder and return a
62    /// handle for communicating with the agent.
63    pub(crate) fn spawn(&self) -> io::Result<Handle> {
64        let create_start = Instant::now();
65
66        // Initialize libcurl, if necessary, on the current thread.
67        //
68        // Note that as of 0.4.30, the curl crate will attempt to do this for us
69        // on the main thread automatically at program start on most targets,
70        // but on other targets must still be initialized on the main thread. We
71        // do this here in the hope that the user builds an `HttpClient` on the
72        // main thread (as opposed to waiting for `Multi::new()` to do it for
73        // us below, which we _know_ is not on the main thread).
74        //
75        // See #189.
76        curl::init();
77
78        let id = NEXT_AGENT_ID.fetch_add(1);
79
80        // Create an I/O selector for driving curl's sockets.
81        let selector = Selector::new()?;
82
83        let (message_tx, message_rx) = async_channel::unbounded();
84
85        let wait_group = WaitGroup::new();
86        let wait_group_thread = wait_group.clone();
87
88        let AgentBuilder {
89            max_connections,
90            max_connections_per_host,
91            connection_cache_size,
92        } = *self;
93
94        // Create a span for the agent thread that outlives this method call,
95        // but rather was caused by it.
96        let agent_span = tracing::debug_span!("agent_thread", id);
97        agent_span.follows_from(tracing::Span::current());
98
99        let waker = selector.waker();
100        let message_tx_clone = message_tx.clone();
101
102        let thread_main = move || {
103            let _enter = agent_span.enter();
104            let mut multi = Multi::new();
105
106            if max_connections > 0 {
107                multi
108                    .set_max_total_connections(max_connections)
109                    .map_err(Error::from_any)?;
110            }
111
112            if max_connections_per_host > 0 {
113                multi
114                    .set_max_host_connections(max_connections_per_host)
115                    .map_err(Error::from_any)?;
116            }
117
118            // Only set maxconnects if greater than 0, because 0 actually means unlimited.
119            if connection_cache_size > 0 {
120                multi
121                    .set_max_connects(connection_cache_size)
122                    .map_err(Error::from_any)?;
123            }
124
125            let agent = AgentContext::new(multi, selector, message_tx_clone, message_rx)?;
126
127            drop(wait_group_thread);
128
129            tracing::debug!("agent took {:?} to start up", create_start.elapsed());
130
131            let result = agent.run();
132
133            if let Err(error) = &result {
134                tracing::error!(?error, "agent shut down with error");
135            }
136
137            result
138        };
139
140        let handle = Handle {
141            message_tx,
142            waker,
143            join_handle: Mutex::new(Some(
144                thread::Builder::new()
145                    .name(format!("isahc-agent-{}", id))
146                    .spawn(thread_main)?,
147            )),
148        };
149
150        // Block until the agent thread responds.
151        wait_group.wait();
152
153        Ok(handle)
154    }
155}

157/// A handle to an active agent running in a background thread.
158///
159/// Dropping the handle will cause the agent thread to shut down and abort any
160/// pending transfers.
161#[derive(Debug)]
162pub(crate) struct Handle {
163    /// Used to send messages to the agent thread.
164    message_tx: Sender<Message>,
165
166    /// A waker that can wake up the agent thread while it is polling.
167    waker: Waker,
168
169    /// A join handle for the agent thread.
170    join_handle: Mutex<Option<thread::JoinHandle<Result<(), Error>>>>,
171}
172
173/// Internal state of an agent thread.
174///
175/// The agent thread runs the primary client event loop, which is essentially a
176/// traditional curl multi event loop with some extra bookkeeping and async
177/// features like wakers.
178struct AgentContext {
179    /// A curl multi handle, of course.
180    multi: curl::multi::Multi,
181
182    /// Used to send messages to the agent thread.
183    message_tx: Sender<Message>,
184
185    /// Incoming messages from the agent handle.
186    message_rx: Receiver<Message>,
187
188    /// Contains all of the active requests.
189    requests: Slab<curl::multi::Easy2Handle<RequestHandler>>,
190
191    /// Indicates if the thread has been requested to stop.
192    close_requested: bool,
193
194    /// A waker that can wake up the agent thread while it is polling.
195    waker: Waker,
196
197    /// This is the poller we use to poll for socket activity!
198    selector: Selector,
199
200    /// A timer we use to keep track of curl's timeouts.
201    timer: Arc<Timer>,
202
203    /// Map of socket registration update requests from the multi handle. A map
204    /// is used to deduplicate multiple updates for the same socket that may
205    /// occur in a single turn of the event loop.
206    socket_updates: Arc<Mutex<HashMap<Socket, SocketEvents, BuildHasherDefault<IntHasher>>>>,
207}
208
209/// A message sent from the main thread to the agent thread.
210#[derive(Debug)]
211enum Message {
212    /// Requests the agent to close.
213    Close,
214
215    /// Begin executing a new request.
216    Execute(EasyHandle),
217
218    /// Request to resume reading the request body for the request with the
219    /// given ID.
220    UnpauseRead(usize),
221
222    /// Request to resume writing the response body for the request with the
223    /// given ID.
224    UnpauseWrite(usize),
225}
226
227#[derive(Debug)]
228enum JoinResult {
229    AlreadyJoined,
230    Ok,
231    Err(Error),
232    Panic,
233}
234
235impl Handle {
236    /// Begin executing a request with this agent.
237    pub(crate) fn submit_request(&self, request: EasyHandle) -> Result<(), Error> {
238        self.send_message(Message::Execute(request))
239    }
240
241    /// Send a message to the agent thread.
242    ///
243    /// If the agent is not connected, an error is returned.
244    fn send_message(&self, message: Message) -> Result<(), Error> {
245        match self.message_tx.try_send(message) {
246            Ok(()) => {
247                // Wake the agent thread up so it will check its messages soon.
248                self.waker.wake_by_ref();
249                Ok(())
250            }
251            Err(_) => match self.try_join() {
252                JoinResult::Err(e) => panic!("agent thread terminated with error: {:?}", e),
253                JoinResult::Panic => panic!("agent thread panicked"),
254                _ => panic!("agent thread terminated prematurely"),
255            },
256        }
257    }
258
259    fn try_join(&self) -> JoinResult {
260        let mut option = self.join_handle.lock().unwrap();
261
262        if let Some(join_handle) = option.take() {
263            match join_handle.join() {
264                Ok(Ok(())) => JoinResult::Ok,
265                Ok(Err(e)) => JoinResult::Err(e),
266                Err(_) => JoinResult::Panic,
267            }
268        } else {
269            JoinResult::AlreadyJoined
270        }
271    }
272}
273
274impl Drop for Handle {
275    fn drop(&mut self) {
276        // Request the agent thread to shut down.
277        if self.send_message(Message::Close).is_err() {
278            tracing::error!("agent thread terminated prematurely");
279        }
280
281        // Wait for the agent thread to shut down before continuing.
282        match self.try_join() {
283            JoinResult::Ok => tracing::trace!("agent thread joined cleanly"),
284            JoinResult::Err(error) => tracing::error!(?error, "agent thread terminated with error"),
285            JoinResult::Panic => tracing::error!("agent thread panicked"),
286            _ => {}
287        }
288    }
289}

291impl AgentContext {
292    fn new(
293        mut multi: Multi,
294        selector: Selector,
295        message_tx: Sender<Message>,
296        message_rx: Receiver<Message>,
297    ) -> Result<Self, Error> {
298        let timer = Arc::new(Timer::new());
299        let socket_updates = Arc::new(Mutex::new(HashMap::with_hasher(Default::default())));
300        let socket_updates_clone = socket_updates.clone();
301
302        multi
303            .socket_function(move |socket, events, _| {
304                // Get mutable access to the map holding pending socket
305                // registration updates that have not been applied yet. Note
306                // that logically this runs on the same thread as the code that
307                // consumes this, so something is very wrong if the lock is
308                // currently being held.
309                let mut socket_updates = socket_updates_clone
310                    .try_lock()
311                    .expect("unexpected socket lock contention");
312
313                // If we have a pending request for this socket that we have not
314                // applied yet, replace it with this new event set. Last word
315                // always wins.
316                socket_updates.insert(socket, events);
317            })
318            .map_err(Error::from_any)?;
319
320        multi
321            .timer_function({
322                let timer = timer.clone();
323
324                move |timeout| match timeout {
325                    Some(timeout) => {
326                        timer.start(timeout);
327                        true
328                    }
329                    None => {
330                        timer.stop();
331                        true
332                    }
333                }
334            })
335            .map_err(Error::from_any)?;
336
337        Ok(Self {
338            multi,
339            message_tx,
340            message_rx,
341            requests: Slab::new(),
342            close_requested: false,
343            waker: selector.waker(),
344            selector,
345            timer,
346            socket_updates,
347        })
348    }
349
350    #[tracing::instrument(level = "trace", skip(self))]
351    fn begin_request(&mut self, mut request: EasyHandle) -> Result<(), Error> {
352        // Prepare an entry for storing this request while it executes.
353        let entry = self.requests.vacant_entry();
354        let id = entry.key();
355        let handle = request.raw();
356
357        // Initialize the handler.
358        request.get_mut().init(
359            id,
360            handle,
361            {
362                let tx = self.message_tx.clone();
363
364                self.waker
365                    .chain(move |inner| match tx.try_send(Message::UnpauseRead(id)) {
366                        Ok(()) => inner.wake_by_ref(),
367                        Err(_) => {
368                            tracing::warn!(id, "agent went away while resuming read for request")
369                        }
370                    })
371            },
372            {
373                let tx = self.message_tx.clone();
374
375                self.waker
376                    .chain(move |inner| match tx.try_send(Message::UnpauseWrite(id)) {
377                        Ok(()) => inner.wake_by_ref(),
378                        Err(_) => {
379                            tracing::warn!(id, "agent went away while resuming write for request")
380                        }
381                    })
382            },
383        );
384
385        // Register the request with curl.
386        let mut handle = self.multi.add2(request).map_err(Error::from_any)?;
387        handle.set_token(id).map_err(Error::from_any)?;
388
389        // Add the handle to our bookkeeping structure.
390        entry.insert(handle);
391
392        Ok(())
393    }
394
395    #[tracing::instrument(level = "trace", skip(self))]
396    fn complete_request(
397        &mut self,
398        token: usize,
399        result: Result<(), curl::Error>,
400    ) -> Result<(), Error> {
401        let handle = self.requests.remove(token);
402        let mut handle = self.multi.remove2(handle).map_err(Error::from_any)?;
403
404        handle.get_mut().set_result(result.map_err(Error::from_any));
405
406        Ok(())
407    }
408
409    /// Polls the message channel for new messages from any agent handles.
410    ///
411    /// If there are no active requests right now, this function will block
412    /// until a message is received.
413    #[tracing::instrument(level = "trace", skip(self))]
414    fn poll_messages(&mut self) -> Result<(), Error> {
415        while !self.close_requested {
416            if self.requests.is_empty() {
417                match block_on(self.message_rx.recv()) {
418                    Ok(message) => self.handle_message(message)?,
419                    _ => {
420                        tracing::warn!("agent handle disconnected without close message");
421                        self.close_requested = true;
422                        break;
423                    }
424                }
425            } else {
426                match self.message_rx.try_recv() {
427                    Ok(message) => self.handle_message(message)?,
428                    Err(async_channel::TryRecvError::Empty) => break,
429                    Err(async_channel::TryRecvError::Closed) => {
430                        tracing::warn!("agent handle disconnected without close message");
431                        self.close_requested = true;
432                        break;
433                    }
434                }
435            }
436        }
437
438        Ok(())
439    }
440
441    #[tracing::instrument(level = "trace", skip(self))]
442    fn handle_message(&mut self, message: Message) -> Result<(), Error> {
443        tracing::trace!("received message from agent handle");
444
445        match message {
446            Message::Close => self.close_requested = true,
447            Message::Execute(request) => self.begin_request(request)?,
448            Message::UnpauseRead(token) => {
449                if let Some(request) = self.requests.get(token) {
450                    if let Err(error) = request.unpause_read() {
451                        // If unpausing returned an error, it is likely because
452                        // curl called our callback inline and the callback
453                        // returned an error. Unfortunately this does not affect
454                        // the normal state of the transfer, so we need to keep
455                        // the transfer alive until it errors through the normal
456                        // means, which is likely to happen this turn of the
457                        // event loop anyway.
458                        tracing::debug!(id = token, ?error, "error unpausing read for request");
459                    }
460                } else {
461                    tracing::warn!(
462                        id = token,
463                        "received unpause request for unknown request token",
464                    );
465                }
466            }
467            Message::UnpauseWrite(token) => {
468                if let Some(request) = self.requests.get(token) {
469                    if let Err(error) = request.unpause_write() {
470                        // If unpausing returned an error, it is likely because
471                        // curl called our callback inline and the callback
472                        // returned an error. Unfortunately this does not affect
473                        // the normal state of the transfer, so we need to keep
474                        // the transfer alive until it errors through the normal
475                        // means, which is likely to happen this turn of the
476                        // event loop anyway.
477                        tracing::debug!(id = token, ?error, "error unpausing write for request");
478                    }
479                } else {
480                    tracing::warn!(
481                        id = token,
482                        "received unpause request for unknown request token",
483                    );
484                }
485            }
486        }
487
488        Ok(())
489    }

491    /// Run the agent in the current thread until requested to stop.
492    fn run(mut self) -> Result<(), Error> {
493        let mut multi_messages = Vec::new();
494
495        // Agent main loop.
496        loop {
497            self.poll_messages()?;
498
499            if self.close_requested {
500                break;
501            }
502
503            // Block until activity is detected or the timeout passes.
504            self.poll()?;
505
506            // Collect messages from curl about requests that have completed,
507            // whether successfully or with an error.
508            self.multi.messages(|message| {
509                if let Some(result) = message.result() {
510                    if let Ok(token) = message.token() {
511                        multi_messages.push((token, result));
512                    }
513                }
514            });
515
516            for (token, result) in multi_messages.drain(..) {
517                self.complete_request(token, result)?;
518            }
519        }
520
521        tracing::debug!("agent shutting down");
522
523        self.requests.clear();
524
525        Ok(())
526    }
527
528    /// Block until activity is detected or a timeout passes.
529    fn poll(&mut self) -> Result<(), Error> {
530        let now = Instant::now();
531        let timeout = self.timer.get_remaining(now);
532
533        // Get the latest timeout value from curl that we should use, limited to
534        // a maximum we chose.
535        let poll_timeout = timeout.map(|t| t.min(WAIT_TIMEOUT)).unwrap_or(WAIT_TIMEOUT);
536
537        // Block until either an I/O event occurs on a socket, the timeout is
538        // reached, or the agent handle interrupts us.
539        if self.selector.poll(poll_timeout)? {
540            // At least one I/O event occurred, handle them.
541            for (socket, readable, writable) in self.selector.events() {
542                tracing::trace!(socket, readable, writable, "socket event");
543                let mut events = Events::new();
544                events.input(readable);
545                events.output(writable);
546                self.multi
547                    .action(socket, &events)
548                    .map_err(Error::from_any)?;
549            }
550        }
551
552        // If curl gave us a timeout, check if it has expired.
553        if self.timer.is_expired(now) {
554            self.timer.stop();
555            self.multi.timeout().map_err(Error::from_any)?;
556        }
557
558        // Apply any requested socket updates now.
559        for (socket, events) in self
560            .socket_updates
561            .try_lock()
562            .expect("unexpected socket lock contention")
563            .drain()
564        {
565            // Curl is asking us to stop polling this socket.
566            if events.remove() {
567                self.selector.deregister(socket)?;
568            } else {
569                let readable = events.input() || events.input_and_output();
570                let writable = events.output() || events.input_and_output();
571
572                self.selector.register(socket, readable, writable)?;
573            }
574        }
575
576        Ok(())
577    }
578}

580impl Drop for AgentContext {
581    fn drop(&mut self) {
582        // Due to a bug present in some versions of curl-rust, dropping an
583        // easy handle after its associated multi handle has already been
584        // dropped can cause a segfault. As a workaround, explicitly drop all
585        // easy handles here first before the multi handle is dropped, if any.
586        //
587        // See https://github.com/sagebind/isahc/issues/459 for more details.
588        self.requests.clear();
589    }
